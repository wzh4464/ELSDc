41static void ellipse2param( double *p, double *param)
42{
43 /* p = [ a, 1/2*b, 1/2*d,
44 1/2*b, c, 1/2*e,
45 1/2*d, 1/2*e, f ]; */
46 double a, b, c, d, e, f;
47 double thetarad, cost, sint, cos_squared, sin_squared, cos_sin;
48 double Ao, Au, Av, Auu, Avv;
49 double tuCentre, tvCentre, wCentre, uCentre, vCentre, Ru, Rv;
50
51 /* check parameters */
52 if( p == NULL ) error("ellipse2param: null input ellipse matrix.");
53 if( param == NULL ) error("ellipse2param: output 'param' should be non-null");
54
55 a = p[0];
56 b = 2*p[1];
57 c = p[4];
58 d = 2*p[2];
59 e = 2*p[5];
60 f = p[8];
61
62 thetarad = 0.5*atan2(b,a-c);
63 cost = cos(thetarad);
64 sint = sin(thetarad);
65 sin_squared = sint * sint;
66 cos_squared = cost * cost;
67 cos_sin = sint * cost;
68 Ao = f;
69 Au = d*cost + e*sint;
70 Av = -d*sint + e*cost;
71 Auu = a*cos_squared + c*sin_squared + b*cos_sin;
72 Avv = a*sin_squared + c*cos_squared - b*cos_sin;
73
74 if( (Auu == 0) || (Avv==0) )
75 {
76 param[0] = 0;
77 param[1] = 0;
78 param[2] = 0;
79 param[3] = 0;
80 param[4] = 0;
81 }
82 else
83 {
84 tuCentre = -Au / (2.*Auu);
85 tvCentre = -Av / (2.*Avv);
86
87 wCentre = Ao-Auu*tuCentre*tuCentre - Avv*tvCentre * tvCentre;
88 uCentre = tuCentre*cost - tvCentre*sint;
89 vCentre = tuCentre*sint + tvCentre*cost;
90
91 Ru = -wCentre / Auu;
92 Rv = -wCentre / Avv;
93
94 if( Ru>0 ) Ru = pow( Ru,0.5);
95 else Ru = -pow(-Ru,0.5);
96
97 if( Rv>0 ) Rv = pow( Rv,0.5);
98 else Rv = -pow(-Rv,0.5);
99
100 param[0] = uCentre;
101 param[1] = vCentre;
102 param[2] = Ru;
103 param[3] = Rv;
104 param[4] = thetarad;
105 }
106}

116static void vgg_conditioner_from_points( double *T, Point *reg, int reg_size )
117{
118 double mx = 0.0, my = 0.0;
119 double qmean = 0.0, Qt = 0.0, val;
120 int i;
121 /* check parameters */
122 if( reg == NULL ) error("vgg_conditioner_from_points: invalid points list.");
123 if( reg_size <= 0 ) error("vgg_conditioner_from_points: invalid size list.");
124
125 /* compute mean point */
126 for( i=0; i<reg_size; i++ )
127 {
128 mx += reg[i].x;
129 my += reg[i].y;
130 }
131 mx /= (double)reg_size; my /= (double)reg_size;
132
133 /* compute mean variance */
134 for( i=0; i<reg_size; i++ )
135 {
136 val = (reg[i].x - mx)*(reg[i].x - mx)+(reg[i].y - my)*(reg[i].y - my);
137 Qt += sqrt(val);
138 }
139 qmean = Qt/(double)reg_size;
140
141 /* build normalization matrix */
142 val = SQRT2/qmean;
143 T[1] = T[3] = 0;
144 T[0] = T[4] = val;
145 T[2] = -val * mx;
146 T[5] = -val * my;
147 T[6] = T[7] = 0;
148 T[8] = 1;
149}

155static void antisym( double *u, double *A )
156{
157 A[0] = A[4] = A[8] = 0;
158 A[1] = -u[2];
159 A[2] = u[1];
160 A[3] = u[2];
161 A[5] = -u[0];
162 A[6] = -u[1];
163 A[7] = u[0];

171static void fit_equations( PImageDouble gradx, PImageDouble grady, double *vgg,
172 Point *reg, int reg_size, double **buff,
173 int *size_buff_max )
174{
175 int i,j;
176 double K[27];
177 double asym[9];
178 int idx;
179 double crosspr[3];
180 double pnormx, pnormy, dirnormx, dirnormy;
181 int addr;
182 /* check parameters */
183 if( (gradx == NULL) || (gradx->data == NULL) )
184 error("fit_equations: invalid input 'gradx' image.");
185 if( (grady == NULL) || (grady->data == NULL) )
186 error("fit_equations: invalid input 'grady' image.");
187 if( vgg == NULL ) error("fit_equations: invalid normalization matrix.");
188 if( reg == NULL ) error("fit_equations: invalid region.");
189 if( reg_size<=0 ) error("fit_equations: invalid region size.");
190 if( *buff == NULL ) error("fit_equations: invalid buffer.");
191
192 /* if buffer too small, allocate twice the memory required at this step */
193 if( reg_size * 24 > *size_buff_max )
194 {
195 *buff = (double*) realloc( *buff,
196 sizeof(double) * 2 * reg_size * 24 );
197 if( *buff == NULL ) error("fit_equations: not enough memory.");
198 *size_buff_max = 2 * reg_size * 24;
199 }
200
201
202 /* compute normalisation matrix */
203 vgg_conditioner_from_points( vgg, reg, reg_size );
204
205 /* compute equation system */
206 for( i=0; i<reg_size; i++)
207 {
208 idx = i*4*6;
209 /* normalise point (pnormx,pnormy) = VGG*(x,y) */
210 pnormx = vgg[0]*reg[i].x + vgg[1]*reg[i].y + vgg[2];
211 pnormy = vgg[3]*reg[i].x + vgg[4]*reg[i].y + vgg[5];
212
213 /* normalise gradient direction (dirnormx,dirnormy) = VGG*(dx,dy) */
214 addr = reg[i].y * gradx->xsize + reg[i].x;
215 dirnormx = -vgg[0] * grady->data[addr] + vgg[1] * gradx->data[addr];
216 dirnormy = -vgg[3] * grady->data[addr] + vgg[4] * gradx->data[addr];
217
218 /* cross product (pnormx,pnormy) x (dirnormx,dirnormy) = tangent line */
219 crosspr[0] = -dirnormy;
220 crosspr[1] = dirnormx;
221 crosspr[2] = pnormx * dirnormy - pnormy * dirnormx;
222
223 /* tangent equation: eq = -transpose(kron(TPts(1:3,i),antisym(l)))*J; */
224 antisym(crosspr,asym);
225
226 for (j=0;j<9;j++) K[j] = asym[j]*pnormx;
227 for (j=0;j<9;j++) K[j+9] = asym[j]*pnormy;
228 for (j=0;j<9;j++) K[j+18] = asym[j];
229
230 (*buff)[idx] = -K[0];
231 (*buff)[idx+1] = -(K[3]+K[9]);
232 (*buff)[idx+2] = -(K[6]+K[18]);
233 (*buff)[idx+3] = -K[12];
234 (*buff)[idx+4] = -(K[15]+K[21]);
235 (*buff)[idx+5] = -K[24];
236
237 (*buff)[idx+6] = -K[1];
238 (*buff)[idx+6+1] = -(K[4]+K[10]);
239 (*buff)[idx+6+2] = -(K[7]+K[19]);
240 (*buff)[idx+6+3] = -K[13];
241 (*buff)[idx+6+4] = -(K[16]+K[22]);
242 (*buff)[idx+6+5] = -K[25];
243
244 (*buff)[idx+12] = -K[2];
245 (*buff)[idx+12+1] = -(K[5]+K[11]);
246 (*buff)[idx+12+2] = -(K[8]+K[20]);
247 (*buff)[idx+12+3] = -K[14];
248 (*buff)[idx+12+4] = -(K[17]+K[23]);
249 (*buff)[idx+12+5] = -K[26];
250
251 /* position equation: eq = transpose(kron(TPts(1:3,i),TPts(1:3,i)))*J; */
252 (*buff)[idx+18] = pnormx * pnormx;
253 (*buff)[idx+19] = 2 * pnormx * pnormy;
254 (*buff)[idx+20] = 2 * pnormx;
255 (*buff)[idx+21] = pnormy * pnormy;
256 (*buff)[idx+22] = 2 * pnormy;
257 (*buff)[idx+23] = 1;
258 }
259}

266static void fit_circle( double *buff, double *vgg, int reg_size, double *param )
267{
268 int i, j, k;
269 double A[16];
270 /* check parameters */
271 if( reg_size<=0 ) error("fit_circle: invalid size.");
272 if( buff == NULL ) error("fit_circle: invalid buffer.");
273 if( vgg == NULL ) error("fit_circle: invalid normalization matrix.");
274 if( param == NULL ) error("fit_circle: param must be non null.");
275
276 /* circle fitting computes less parameters, so the equation matrix has
277 only four columns, not six; modify accordingly */
278 for( i=0; i<reg_size*4*6; i+=6 )
279 {
280 buff[i] = buff[i] + buff[i+3];
281 buff[i+1] = buff[i+2];
282 buff[i+2] = buff[i+4];
283 buff[i+3] = buff[i+5];
284 }
285 /* A = EQ'*EQ; */
286 for( i=0; i<16; i++) A[i] = 0.0;
287
288 for( i=0; i<4; i++ )
289 for( j=0; j<4; j++)
290 for(k=0; k<4*reg_size; k++)
291 A[i*4+j] += buff[k*6+i] * buff[k*6+j];
292
293 /* lapack call to solve linear system */
294 lap_eig( A, 4 );
295
296 double s[9];
297 s[0] = s[4] = A[0];
298 s[1] = s[3] = 0;
299 s[2] = s[6] = A[1];
300 s[5] = s[7] = A[2];
301 s[8] = A[3];
302
303 /* apply inverse(normalisation matrix) */
304 /* C = T'*[ x(1),0,x(2); 0,x(1),x(3) ; x(2),x(3),x(4)]*T;*/
305
306 double C[9];
307 C[0] = vgg[0]*vgg[0]*s[0] + vgg[3]*vgg[3]*s[4];
308 C[1] = vgg[0]*vgg[1]*s[0] + vgg[3]*vgg[4]*s[4];
309 C[2] = vgg[0]*vgg[2]*s[0] + vgg[3]*vgg[5]*s[4]+vgg[0]*s[2] + vgg[3]*s[5];
310
311 C[3] = vgg[0]*vgg[1]*s[0] + vgg[3]*vgg[4]*s[4];
312 C[4] = vgg[1]*vgg[1]*s[0] + vgg[4]*vgg[4]*s[4];
313 C[5] = vgg[1]*vgg[2]*s[0] + vgg[4]*vgg[5]*s[4] + vgg[1]*s[2] + vgg[4]*s[5];
314
315 C[6] = vgg[0]*vgg[2]*s[0] + vgg[0]*s[6] + vgg[3]*vgg[5]*s[4] + vgg[3]*s[7];
316 C[7] = vgg[1]*vgg[2]*s[0] + vgg[1]*s[6] + vgg[4]*vgg[5]*s[4] + vgg[4]*s[7];
317 C[8] = vgg[2]*vgg[2]*s[0] + vgg[2]*s[6] + vgg[5]*vgg[5]*s[4] + vgg[5]*s[7]+
318 vgg[2]*s[2] + vgg[5]*s[5] + s[8];
319
320 ellipse2param( C, param );
321}

327static void fit_ellipse( double *buff, double *vgg, int reg_size,
328 double *param )
329{
330 int i,j,k;
331 double A[36];
332 /* check parameters */
333 if( reg_size <= 0 ) error("fit_ellipse: invalid size.");
334 if( buff == NULL ) error("fit_ellipse: invalid buffer.");
335 if( vgg == NULL ) error("fit_ellipse: invalid normalization matrix.");
336 if( param == NULL ) error("fit_ellipse: param must be non null.");
337
338 /* A = EQ'*EQ; */
339 for( i=0; i<36; i++ ) A[i] = 0.0;
340
341 for( i=0; i<6; i++ )
342 for( j=0; j<6; j++ )
343 for( k=0; k<4*reg_size; k++ )
344 A[i*6+j] += buff[k*6+i] * buff[k*6+j];
345
346 /* lapack call to solve linear system */
347 lap_eig( A, 6 );
348
349 double s[9];
350 s[0] = A[0];
351 s[1] = s[3] = A[1];
352 s[2] = s[6] = A[2];
353 s[4] = A[3];
354 s[5] = s[7] = A[4];
355 s[8] = A[5];
356
357 /* apply inverse(normalisation matrix) */
358 /* C = T'*[ x(1),x(2),x(3); x(2),x(4),x(5) ; x(3),x(5),x(6)]*T; */
359
360 double C[9];
361 C[0] = vgg[0]*vgg[0]*s[0] + vgg[0]*vgg[3]*s[3] +
362 vgg[0]*vgg[3]*s[1] + vgg[3]*vgg[3]*s[4];
363 C[1] = vgg[0]*vgg[1]*s[0] + vgg[1]*vgg[3]*s[3] +
364 vgg[0]*vgg[4]*s[1] + vgg[3]*vgg[4]*s[4];
365 C[2] = vgg[0]*vgg[2]*s[0] + vgg[2]*vgg[3]*s[3] +
366 vgg[0]*vgg[5]*s[1] + vgg[3]*vgg[5]*s[4] + vgg[0]*s[2] + vgg[3]*s[5];
367
368 C[3] = vgg[0]*vgg[1]*s[0] + vgg[0]*vgg[4]*s[3] +
369 vgg[1]*vgg[3]*s[1] + vgg[3]*vgg[4]*s[4];
370 C[4] = vgg[1]*vgg[1]*s[0] + vgg[1]*vgg[4]*s[3] +
371 vgg[1]*vgg[4]*s[1] + vgg[4]*vgg[4]*s[4];
372 C[5] = vgg[1]*vgg[2]*s[0] + vgg[2]*vgg[4]*s[3] +
373 vgg[1]*vgg[5]*s[1] + vgg[4]*vgg[5]*s[4] + vgg[1]*s[2] + vgg[4]*s[5];
374
375 C[6] = vgg[0]*vgg[2]*s[0] + vgg[0]*vgg[5]*s[3] +
376 vgg[0]*s[6] + vgg[2]*vgg[3]*s[1] + vgg[3]*vgg[5]*s[4] + vgg[3]*s[7];
377 C[7] = vgg[1]*vgg[2]*s[0] + vgg[1]*vgg[5]*s[3] +
378 vgg[1]*s[6] + vgg[2]*vgg[4]*s[1] + vgg[4]*vgg[5]*s[4] + vgg[4]*s[7];
379 C[8] = vgg[2]*vgg[2]*s[0] + vgg[2]*vgg[5]*s[3] + vgg[2]*s[6] +
380 vgg[2]*vgg[5]*s[1] + vgg[5]*vgg[5]*s[4] + vgg[5]*s[7] + vgg[2]*s[2] +
381 vgg[5]*s[5] + s[8];
382
383 ellipse2param(C,param);
384}
