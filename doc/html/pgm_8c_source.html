42static void skip_whites_and_comments( FILE *f )
43{
44 int c;
45 do
46 {
47 while( isspace( c = getc(f) ) ); /* skip spaces */
48 if( c == '#') /* skip comments */
49 while( (c != '\n') && (c != '\r') && (c != EOF) )
50 c = getc(f);
51 }
52
53 while( (c == '#') || (isspace(c)) );
54 if( (c != EOF) && (ungetc( c, f ) == EOF) )
55 error("skip_whites_and_comments: unable to 'ungetc' in reading PGM file.");
56}

62static unsigned int get_num( FILE *f )
63{
64 unsigned int num;
65 int c;
66
67 while( isspace( c = getc(f) ) ); /* skip spaces */
68 if( !isdigit(c) ) error("Error: corrupted PGM file.");
69
70 num = (unsigned int) (c - '0');
71 while( isdigit( c = getc(f) ) ) num = 10 * num + c - '0';
72 if( (c != EOF) && (ungetc( c, f ) == EOF) )
73 error("get_num: unable to 'ungetc' while reading PGM file.");
74
75 return num;
76}

83PImageDouble read_pgm_image_double( char *name )
84{
85 FILE *f;
86 int c, bin = FALSE;
87 unsigned int xsize, ysize, depth, x, y;
88 PImageDouble image;
89
90 /* open file */
91 f = fopen( name,"rb");
92 if( f == NULL )
93 error("read_pgm_image_double: unable to open input image file.");
94
95 /* read header */
96 if( getc(f) != 'P') error("read_pgm_image_double: not a PGM file.");
97 if( (c = getc(f) ) == '2') bin = FALSE;
98 else if( c == '5') bin = TRUE;
99 else error("read_pgm_image_double: not a PGM file.");
100
101 skip_whites_and_comments(f);
102 xsize = get_num(f); /* X size */
103 skip_whites_and_comments(f);
104 ysize = get_num(f); /* Y size */
105 skip_whites_and_comments(f);
106 depth = get_num(f); /* depth */
107
108 if( depth==0 )
109 fprintf( stderr,"Warning: depth=0, probably invalid PGM file.\n");
110
111 /* white before data */
112 if( !isspace( c = getc(f) ) )
113 error("read_pgm_image_double: corrupted PGM file.");
114
115 /* get memory */
116 image = new_PImageDouble( xsize, ysize );
117
118 /* read data */
119 for( y=0; y<ysize; y++)
120 for( x=0; x<xsize; x++)
121 image->data[ x + y * xsize ] = bin ? (double) getc(f)
122 : (double) get_num(f);
123
124 /* close file if needed */
125 if( (f != stdin) && (fclose(f) == EOF) )
126 error("read_pgm_image_double: unable to close file.");
127
128 return image;
129}

136void write_pgm_image_int( int *image, int xsize, int ysize, char *name )
137{
138 FILE *f;
139 int x, y, n, v, max, min;
140
141 /* check input */
142 if( (image == NULL) || (xsize <= 0) || (ysize <= 0) )
143 error("write_pgm_image_int: invalid input image.");
144
145 /* check min and max values */
146 max = min = 0;
147 for( y=0; y<ysize; y++ )
148 for( x=0; x<xsize; x++ )
149 {
150 v = image[ x + y * xsize ];
151 if( v > max ) max = v;
152 if( v < min ) min = v;
153 }
154
155 if( min < 0 ) fprintf( stderr,
156 "Warning: write_pgm_image_int: negative values in '%s'.\n", name );
157 if( max > 65535 ) fprintf( stderr,
158 "Warning: write_pgm_image_int: values exceeding 65535 in '%s'.\n", name );
159
160 /* open file */
161 if( strcmp( name,"-") == 0 ) f = stdout;
162 else f = fopen( name,"w");
163 if( f == NULL )
164 error("write_pgm_image_int: unable to open output image file.");
165
166 /* write header */
167 fprintf( f,"P2\n");
168 fprintf( f,"%d %d\n", xsize, ysize );
169 fprintf( f,"%d\n", max);
170
171 /* write data */
172 for( n=0, y=0; y<ysize; y++ )
173 for( x=0; x<xsize; x++ )
174 {
175 fprintf( f,"%d ", image[ x + y * xsize ] );
176 if( ++n == 8 ) /* lines should not be longer than 70 characters */
177 {
178 fprintf( f,"\n");
179 n = 0;
180 }
181 }
182
183 /* close file if needed */
184 if( (f != stdout) && (fclose(f) == EOF) )
185 error("write_pgm_image_int: unable to close file.");
186}
