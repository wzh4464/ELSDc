51#define LOG10_1_2_M_2__PI 0.39908993418
52#define LOG10_EXP1 0.43429448190
53static double NFAc( int n, double k, double logNT )
54{
55 /* check parameters */
56 if( n < 0) error("NFAc: 'n' must be strictly positive.");
57 if( n==0 ) return -logNT;
58 double logNFAC = logNT;
59
60 k = max(k,0.01);
61 logNFAC = logNT + n * log10(k) - LOG10_1_2_M_2__PI -
62 (n+0.5) * log10(n) + n * LOG10_EXP1 ;
63
64 return -logNFAC;
65}

78static void rect_count_error( PImageDouble angles, PImageInt used,
79 Rectangle *rec, int *pts, double *err,
80 Point *buff, int *size_buff )
81{
82 RectIter *i;
83 int adr;
84 double theta;
85
86 /* check parameters */
87 if( (angles == NULL) || (angles->data == NULL) )
88 error("rect_count_error: invalid input 'angles' image.");
89 if( (used == NULL) || (used->data == NULL) )
90 error("rect_count_error: invalid input 'used' image.");
91 if( rec == NULL ) error("rect_count_error: invalid input rectangle.");
92 if( buff == NULL ) error("rect_count_error: invalid buffer.");
93 if( size_buff == NULL )
94 error("rect_count_error: size_buff must be non null.");
95
96 theta = rec->theta - M_1_2_PI;
97 *pts = 0; *err = 0.0;
98
99 /* iterate through rectangle and accumulate per-pixel normalised error */
100 for( i=ini_RectIter(rec); !end_RectIter(i); inc_RectIter(i) )
101 if( in_image( i->x, i->y, used->xsize, used->ysize ) )
102 {
103 ++(*pts);
104 adr = (i->y) * used->xsize + (i->x);
105 if( (used->data[adr] == USED) || (angles->data[adr] == NOTDEF) )
106 (*err) += 1.0;
107 else
108 (*err) += norm_angle_diff( angles->data[adr], theta );
109 /* store points in buffer */
110 buff[*size_buff].x = i->x;
111 buff[*size_buff].y = i->y;
112 ++(*size_buff);
113 }
114 free_RectIter(i);
115}

122static void get_seg( PImageDouble angles, PImageDouble gradmag,
123 PImageInt used, PolyRect *poly, Point *reg,
124 int reg_size, double prec )
125{
126 double sumdx = 0.0, sumdy = 0.0;
127 double reg_angle;
128 int i;
129 int adr;
130 Rectangle rec;
131 /* check parameters */
132 if( (angles == NULL) || (angles->data == NULL) )
133 error("get_seg: invalid input 'angles' image.");
134 if( (gradmag == NULL) || (gradmag->data == NULL) )
135 error("get_seg: invalid input 'angles' image.");
136 if( (used == NULL) || (used->data == NULL) )
137 error("get_seg: invalid input 'used' image.");
138 if( poly == NULL ) error("get_seg: invalid input polygon.");
139 if( reg == NULL ) error("get_seg: invalid input region.");
140 if( reg_size <= 0 ) error("get_seg: 'reg_size' must be strictly positive.");
141
142 /* compute the mean angle of the whole region */
143 for( i=0; i<reg_size; i++ )
144 {
145 adr = reg[i].x + reg[i].y * angles->xsize;
146 sumdx += cos( angles->data[adr] );
147 sumdy += sin( angles->data[adr] );
148 }
149 reg_angle = atan2( sumdy, sumdx );
150
151 /* estimate rectangle */
152 region2rect( gradmag, reg, 0, reg_size, reg_angle, &rec, prec );
153
154 /* add rectangle to polygon */
155 poly->dim++;
156 add_rect_to_polyrect( poly, &rec );
157}

170static void poly_count_error( PImageDouble angles, PImageInt used,
171 PolyRect *poly, double polywmin, double polywmax,
172 int *pts, double *err, Point *buff,
173 int *size_buff )
174{
175 int pts_rect;
176 double err_rect;
177 int i;
178 /* check parameters */
179 if( (angles == NULL) || (angles->data == NULL) )
180 error("poly_count_error: invalid input 'angles' image.");
181 if( (used == NULL) || (used->data == NULL) )
182 error("poly_count_error: invalid input 'used' image.");
183 if( poly == NULL ) error("poly_count_error: invalid input polygon.");
184 if( buff == NULL ) error("poly_count_error: invalid buffer.");
185 if( size_buff == NULL )
186 error("poly_count_error: size_buff must be non null.");
187
188 *pts = 0; *err = 0.0;
189 *size_buff = 0;
190
191 /* Scan every rectangle in the polygon and count the additive error.
192 We must consider the same widths wmin and wmax for all rectangles. */
193 for( i=0; i<poly->dim; i++)
194 {
195 poly->rectlist[i].wmin = polywmin;
196 poly->rectlist[i].wmax = polywmax;
197 rect_count_error( angles, used, &(poly->rectlist[i]), &pts_rect,
198 &err_rect, buff, size_buff );
199 (*pts) += pts_rect;
200 (*err) += err_rect;
201 }
202}

236static double poly_improve( PImageDouble angles, PImageInt used, PolyRect *poly,
237 Point **local_buff, int *size_local_buff,
238 Point **tmp_buff, double logNT_poly )
239{
240 double delta = 0.5;
241 double log_nfa,log_nfa_new;
242
243 int size_tmp_buff;
244 double wmin,wmax;
245
246 /* check parameters */
247 if( (angles == NULL) || (angles->data == NULL) )
248 error("poly_improve: invalid input 'angles' image.");
249 if( (used == NULL) || (used->data == NULL) )
250 error("poly_improve: invalid input 'used' image.");
251 if( poly == NULL ) error("poly_improve: invalid input polygon.");
252 if( *local_buff == NULL ) error("poly_improve: invalid local buffer.");
253 if( size_local_buff == NULL )
254 error("poly_improve: 'size_local_buff' must be non null.");
255 if( *tmp_buff == NULL ) error("poly_improve: invalid tmp buffer.");
256
257 *size_local_buff = 0;
258 wmax = poly->wmax;
259 wmin = poly->wmin;
260 /* check if polygon valid: wmin and wmax must have opposite sign */
261 if ( wmin*wmax >0 ) return -logNT_poly;
262
263 log_nfa = poly_nfa( angles, used, poly, wmin, wmax, *local_buff,
264 size_local_buff, logNT_poly );
265
266 /* try to improve the polygon, by reducing width; first reduce wmax */
267 /*for( n=0; n<10; n++ )
268 {*/
269 while( wmax - delta > 0.0)
270 {
271 wmax -= delta;
272 size_tmp_buff = 0;
273
274 log_nfa_new = poly_nfa( angles, used, poly, wmin, wmax, *tmp_buff,
275 &size_tmp_buff, logNT_poly );
276 if( log_nfa_new > log_nfa )
277 {
278 log_nfa = log_nfa_new;
279 /* switch addresses of the pointers to the local and tmp buffer */
280 /*swap_ptr_pts( local_buff, tmp_buff );
281 (*size_local_buff) = size_tmp_buff;*/
282 poly->wmax -= delta;
283 }
284 }
285 /* } */
286
287 /* now try to reduce dmin */
288 /* for( n=0; n<10; n++ )
289 { */
290 while( wmin + delta < 0.0)
291 {
292 wmin += delta;
293 size_tmp_buff = 0;
294
295 log_nfa_new = poly_nfa( angles, used, poly, wmin, poly->wmax,
296 *tmp_buff, &size_tmp_buff, logNT_poly );
297 if( log_nfa_new > log_nfa )
298 {
299 log_nfa = log_nfa_new;
300 /* switch addresses of the pointers to the local and tmp buffer */
301 /* swap_ptr_pts( local_buff, tmp_buff );
302 (*size_local_buff) = size_tmp_buff; */
303 poly->wmin += delta;
304 }
305 }
306 /* } */
307 return log_nfa;
308}

321static void circ_count_error( PImageDouble angles, PImageInt used, Ring *cring,
322 int x, int y, int grad_dir, Point *buff,
323 int *size_buff, int label, int *pts, double *err )
324{
325 int xx, yy, i, adr;
326 double theta;
327
328 /* check parameters */
329 if( (angles == NULL) || (angles->data == NULL) )
330 error("circ_count_error: invalid input 'angles' image.");
331 if( (used == NULL) || (used->data == NULL) )
332 error("circ_count_error: invalid input 'used' image.");
333 if( cring == NULL ) error("circ_count_error: invalid input ring.");
334 if( buff == NULL ) error("circ_count_error: invalid local buffer.");
335 if( size_buff == NULL )
336 error("circ_count_error: 'size_local_buff' must be non null.");
337 if( label <= 1 ) error("circ_count_error: forbidden label value.");
338
339 *pts = 0; *err = 0.0;
340
341 /* To compute the error, visit each pixel in the ring in a region growing
342 manner */
343
344 /* First make sure the seed point (x,y) is inside the ring; if not, visit
345 neighbours in a 3x3 neighbourhood to find a point inside the ring */
346 *size_buff = 0;
347 if( is_in_circ_ring( cring, x, y ) )
348 {
349 buff[0].x = x;
350 buff[0].y = y;
351 used->data[ buff[0].y * used->xsize + buff[0].x ] = label;
352 *size_buff = 1;
353 }
354 else
355 {
356 for( xx=x-2; xx<=x+2; xx++ )
357 {
358 for( yy=y-2; yy<=y+2; yy++ )
359 if( in_image( xx, yy, used->xsize, used->ysize ) && is_in_circ_ring( cring, xx, yy ) )
360 {
361 buff[0].x = xx;
362 buff[0].y = yy;
363 used->data[ buff[0].y * used->xsize + buff[0].x ] = label;
364 *size_buff = 1;
365 break;
366 }
367 if( *size_buff == 1 ) break;
368 }
369 }
370 if( *size_buff == 0 ) return;
371
372 for( i=0; i < *size_buff; i++ )
373 for( xx=buff[i].x-1; xx<=buff[i].x+1; xx++ )
374 for( yy=buff[i].y-1; yy<=buff[i].y+1; yy++ )
375 if( in_image( xx, yy, used->xsize, used->ysize ) )
376 {
377 adr = xx + yy * used->xsize;
378 /* check if the point has already been used or visited */
379 if( (used->data[adr] != USED) && (used->data[adr] != label) )
380 {
381 /* check if point inside the circle ring */
382 if( is_in_circ_ring( cring, xx, yy ) )
383 {
384 /* check if gradient converges or diverges when computing
385 the ideal angle of a point on the circle */
386 theta = atan2( (double)yy - cring->cy,
387 (double)xx - cring->cx );
388 if( grad_dir == 0 )
389 {
390 if( theta > 0 ) theta = -(M_PI-theta);
391 else theta = M_PI + theta;
392 }
393 (*err) += norm_angle_diff( angles->data[adr], theta );
394 used->data[adr] = label;
395 buff[*size_buff].x = xx;
396 buff[*size_buff].y = yy;
397 ++(*size_buff);
398 }
399 }
400 }
401 *pts = *size_buff;
402}

699 }
700 }
701 /*} */

703 /* now try to reduce wmin */
704 copy_ring( ering, &r );
705 /*for( n=0; n<10; n++ )
706 { */
707 while( r.wmin + delta < 0.0)
708 {
709 r.wmin += delta;
710 size_tmp_buff = 0;
711 (*label)++;
712
713 log_nfa_new = ell_ring_nfa( angles, used, &r, foci, x, y, grad_dir,

718 ering->wmin += delta;

721 /*swap_ptr_pts( local_buff, tmp_buff );
722 (*size_local_buff) = size_tmp_buff; */
723 }
724 }
725 /* } */
726 if ( ! check_ell_ring(ering) ) return -logNT_ell;
727 return log_nfa;

780void ELSDc( PImageDouble in, int *ell_count, Ring **ell_out, int **ell_labels,
781 int *poly_count, Polygon **poly_out, int **poly_labels,
782 PImageInt out )
783{
784 double ang_th = 22.5; /* gradient angle tolerance in degrees */
785 double prec; /* radian precision */
786 double p;
787 double rho = 5.0; /* pixels with gradient magnitude under this
788 threshold are not considered, as their (weak)
789 orientation might be too affected by noise */
790 int grad_dirc; /* direction of the gradient wrt center of the
791 circle */
792 int grad_dire; /* direction of the gradient wrt center of the
793 ellipse */
794 double density_th = 0.7; /* minimum density */
795 void *mem_p; /* parameters used for sorting */
796 int n_bins = 1024; /* (binning) pixels according to */
797 double max_grad = 255.0; /* to their gradient magnitude */
798 CoordList *list_p; /* list of sorted pixels */
799 unsigned int xsize,ysize; /* image size */
800 double mlog10eps = 0.0; /* minus log of the number of accepted false
801 positives ( eps=1 ) */
802 int min_size_seg; /* minimum size of a region that could contain a
803 meaningful segment */
804 int min_size_ell; /* minimum size of a region that could contain a
805 meaningful ellipse */
806 int min_size_poly; /* minimum size of a region that could contain a
807 meaningful ellipse */
808 double msize; /* max value allowed for circle/ellipse radius */
809 double logNT_ell; /* log of the number of tests for elliptical arcs */
810 double logNT_seg; /* log of the number of tests for line segments */
811 double logNT_poly; /* log of the number of tests for polygons */
812 double log_nfa, best_nfa; /* NFA values using the continuous */
813 void *best_feature; /* pointer to best feature found so far */
814 char best_type = '0'; /* type of best feature; 'p' for polygon,
815 'e' for ellipse */
816 PImageDouble imgauss; /* smooth version of the original image, used during
817 candidate selection */
818 PImageDouble angles; /* gradient angles smooth image */
819 PImageDouble angles0; /* gradient angles original image */
820 PImageDouble gradmag; /* gradient magnitude */
821 PImageDouble gradx, grady;/* gradient orientations on Ox and Oy */
822 PImageInt used; /* image to mark already used points */
823 PolyRect *poly; /* parameters of polygon as list of rectangles */
824 PolyRect *seg; /* one rectangle is a particular polygon */
825 Ring ering, cring; /* ring parameters */
826 double *buff_fit; /* buffer zone needed for fitting */
827 int size_buff_fit; /* size of the currently allocated memory for
828 the buffer */
829 Point *reg; /* list of points gathered during curve growing */
830 int reg_size;
831 Point *best_buff; /* buffer to store the points of the most meaningful
832 primitive found so far (among segment, polygon,
833 ellipse, circle) in one iteration */
834 int size_best_buff;
835 Point *new_buff; /* buffer to store the points of a new primitive;
836 if the new primitive is more meaningful than
837 the most meaningful primitive found so far,
838 deem new primitive as most meaningful by
839 switching the addresses with best_buff */
840 int size_new_buff;
841 Point *tmp_buff; /* buffer to store the points of a new version of a
842 primitive, computed during improve operation;
843 if the new version is more meaningful than
844 the most meaningful version found so far,
845 deem new version as most meaningful by
846 switching the addresses with new_buff */
847 int ell_count_max = 0;
848 int poly_count_max = 0;
849 int adr;
850 double cparam[5];
851 double eparam[5];
852 double pext1[2], pext2[2], ang0;
853 int label = 2; /* 0 is NOTUSED, 1 is USED, so 'label' must go
854 upward from 2 */
855 int label_meaningful = 0;
856 double spir;
857 double foci[4];
858 /*int i;*/
859
860 /* check parameters */
861 if( (in == NULL) || (in->data == NULL) ) error("ELSDc: invalid input image.");
862 if( (out == NULL) || (out->data == NULL) )
863 error("ELSDc: invalid output label image.");
864
865 /* allocate and init variables */
866 prec = M_PI*ang_th/180.0;
867 p = ang_th/180.0;
868 xsize = in->xsize;
869 ysize = in->ysize;
870
871 /* allocate matrix to store elements of the design matrix used in conic
872 fitting */
873 buff_fit = (double*) malloc ( sizeof (double) );
874 size_buff_fit = 1;
875
876 /* allocate buffers to store point coordinates */
877 reg = (Point *) malloc( xsize * ysize * sizeof(Point) );
878 if( reg == NULL ) error("ELSDc: not enough memory.");
879 best_buff = (Point *) malloc( xsize * ysize * sizeof(Point) );
880 if( best_buff == NULL ) error("ELSDc: not enough memory.");
881 new_buff = (Point *) malloc( xsize * ysize * sizeof(Point) );
882 if( new_buff == NULL ) error("ELSDc: not enough memory.");
883 tmp_buff = (Point *) malloc( xsize * ysize * sizeof(Point) );
884 if( tmp_buff == NULL ) error("ELSDc: not enough memory.");
885
886 used = new_PImageInt_ini( xsize, ysize, NOTUSED );
887
888 /* Number of tests for ellipse arcs. We must count (3*xsize)*(3*ysize)
889 possible values for the center, as center can be outside the image too,
890 (xsize*ysize) possible values for the two axes, sqrt(xsize*ysize) possible
891 values for orientation, sqrt(xsize*ysize) possible values for width,
892 (xsize*ysize) possible values for delimiting angles. This makes
893 9*(xsize*ysize)^4. Since we have polygons and ellipses, i.e. two families,
894 we must multiply this by 2. So NT_ell = 2*9*(xsize*ysize)^4.
895 In log, we get: */
896 logNT_ell = 4.0 * ( log10( (double)xsize ) + log10( (double)ysize ) ) +
897 LOG10_9 + LOG10_2;
898 min_size_ell = (int)( (-logNT_ell-mlog10eps) / log10(p) );
899
900 logNT_seg = 2.5 * (log10((double)xsize) + log10((double)ysize) )
901 + LOG10_2;
902 min_size_seg = (int)( (-logNT_seg - mlog10eps) / log10(p) );
903
904 /* perform gaussian smoothing */
905 imgauss = gaussian_sampler( in, 1.0, 0.6 );
906
907 /* compute gradient magnitude and orientation for the smooth image */
908 img_gradient_sort( imgauss, rho, &list_p, &mem_p, n_bins, max_grad, &angles,
909 &gradmag, &gradx, &grady );
910
911 /* compute gradient orientation for the original image */
912 angles0 = img_gradient_angle( in, rho );
913
914 /* input image not needed any more; free it */
915 free_PImageDouble(in);
916
917 msize = max(xsize,ysize);
918
919 poly = new_polyrect();
920 seg = new_polyrect();
921
922
923 /* begin primitive detection */
924 for( ; list_p; list_p = list_p->next )
925 {
926 adr = list_p->y * used->xsize + list_p->x;
927 /* only pixels that haven't been used already can be seed pixels */
928 if( used->data[adr] == USED ) continue;
929
930 /* only pixels with defined gradient can be seed pixels */
931 if( angles->data[adr] == NOTDEF ) continue;
932
933 /* only pixels that are local maxima in the image can be seed pixels */
934 /*if( !local_max( imgauss, list_p->x, list_p->y ) ) continue; */
935
936 /* init intra-iteration variables */
937 log_nfa = best_nfa = mlog10eps;
938 best_feature = NULL;
939 best_type = '0';
940 size_best_buff = size_new_buff = 0;
941 label++;
942 spir = 0.0;
943 clear_polyrect(poly);
944 clear_polyrect(seg);
945
946 /* set initial seed point */
947 reg[0].x = list_p->x;
948 reg[0].y = list_p->y;
949 reg_size = 1;
950
951 /* Gather points aligned along a convex polygon */
952 if( !curve_grow( gradmag, angles, used, reg, &reg_size, density_th,
953 prec, poly, &label, pext2, pext1, &spir ) )
954 continue;
955
956
957 /* Validate polygon */
958 /* Compute number of tests for polygon; it must be done for each
959 polygon, as it depends on the number of segments in the
960 polygon */
961 logNT_poly = ( poly->dim + 1.5 ) * ( log10((double)xsize ) +
962 log10( (double)ysize) ) + ( poly->dim+1.0 ) * LOG10_2;
963 /* polygon's best validation score */
964 min_size_poly =(int)((-logNT_poly - mlog10eps)/log10(p));
965
966 if( reg_size > min_size_poly )
967 {
968 best_nfa = poly_improve( angles0, used, poly, &best_buff,
969 &size_best_buff, &tmp_buff, logNT_poly );
970 best_feature = (void*)poly;
971 best_type = 'p';
972 }
973
974 /* Estimate a single segment on the entire set and validate;
975 this prevents from considering very small regions at the ends
976 of a large segment as new segments. */
977 get_seg( angles, gradmag, used, seg, reg, reg_size, prec );
978 log_nfa = poly_improve( angles0, used, seg, &new_buff,
979 &size_new_buff, &tmp_buff, logNT_seg );
980
981 /* if more meaningful than previous, update best feature */
982 if( log_nfa > best_nfa && size_new_buff>min_size_seg )
983 {
984 best_nfa = log_nfa;
985 best_feature = (void*)seg;
986 best_type = 'p';
987 swap_ptr_pts( &best_buff, &new_buff );
988 size_best_buff = size_new_buff;
989 }
990
991 /* Estimate circle and ellipse parameters (centre, axes, orientation)
992 on the gathered pixels */
993 if( reg_size > min_size_ell )
994 conic_fit( gradx, grady, reg, reg_size, &buff_fit, &size_buff_fit,
995 cparam, eparam );
996
997 /* Get parameters of the circle ring and compute validation score */
998 ang0 = angles->data[ reg[0].y*angles->xsize + reg[0].x ];
999 if( get_ring( reg, reg_size, ang0, cparam, pext1, pext2, CIRCLE,
1000 spir, &cring, &grad_dirc, foci, msize ) )
1001 {
1002 log_nfa = circ_ring_improve( angles0, used, &cring, reg[0].x,
1003 reg[0].y, grad_dirc, &new_buff,
1004 &size_new_buff, &tmp_buff, &label,
1005 logNT_ell );
1006 /* if more meaningful than previous, update best feature */
1007 if( log_nfa > best_nfa && size_new_buff>min_size_ell )
1008 {
1009 best_nfa = log_nfa;
1010 best_feature = (void*)(&cring);
1011 best_type = 'c';
1012 swap_ptr_pts( &best_buff, &new_buff );
1013 size_best_buff = size_new_buff;
1014 }
1015 } /* ! get cring */
1016
1017 /* Get parameters of the ellipse ring and compute validation score */
1018 if( get_ring( reg, reg_size, ang0, eparam, pext1, pext2, ELLIPSE,
1019 spir, &ering, &grad_dire, foci, msize ) )
1020 {
1021 log_nfa = ell_ring_improve( angles0, used, &ering, foci, reg[0].x,
1022 reg[0].y, grad_dire, &new_buff,
1023 &size_new_buff, &tmp_buff, &label,
1024 logNT_ell );
1025 /* if more meaningful than previous, update best feature */
1026 if( log_nfa > best_nfa && size_new_buff>min_size_ell )
1027 {
1028 best_nfa = log_nfa;
1029 best_feature = (void*)(&ering);
1030 best_type = 'c';
1031 swap_ptr_pts( &best_buff, &new_buff );
1032 size_best_buff = size_new_buff;
1033 }
1034 } /* ! get ering */
1035
1036 /* At this point, we have the most meaninful feature at this iteration.
1037 If it is meaningful compared to eps, then store */
1038 if( best_nfa <= mlog10eps ) continue;
1039 label_meaningful++;
1040 if( best_type == 'c' )
1041 {
1042 (*ell_count)++;
1043 add_ell_out( ell_out, (Ring*)best_feature, *ell_count, &ell_count_max,
1044 ell_labels, label_meaningful );
1045 /*Ring *rr = (Ring*)best_feature;*/
1046 /* free(rr); */
1047 }
1048 else if( best_type == 'p' )
1049 {
1050 (*poly_count)++;
1051 add_poly_out( poly_out, (PolyRect*)best_feature, *poly_count,
1052 &poly_count_max, poly_labels, label_meaningful );
1053 }
1054
1055
1056 /* Mark as USED the pixels of the best feature in 'used' */
1057 mark_img_pts( used, best_buff, 0, size_best_buff, USED );
1058
1059 /* Mark label of the best feature in output label image */
1060 mark_img_pts( out, best_buff, 0, size_best_buff, label_meaningful );
1061
1062 } /* ! for loop */
1063
1064 /* free useless structures and arrays */
1065 free_PImageDouble(imgauss);
1066 free_PImageDouble(gradx);
1067 free_PImageDouble(grady);
1068 free_PImageDouble(gradmag);
1069 free_PImageDouble(angles);
1070 free_PImageDouble(angles0);
1071 free_PImageInt(used);
1072 free(reg);
1073 free(best_buff);
1074 free(new_buff);
1075 free(tmp_buff);
1076 free(buff_fit);
1077 free(poly->rectlist); free(poly);
1078 free(seg->rectlist); free(seg);
1079 free(mem_p);
1080}
