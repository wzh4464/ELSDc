37FILE *init_svg( char *filename, unsigned int xsize, unsigned int ysize )
38{
39 FILE *fsvg;
40
41 /* open file */
42 fsvg = fopen( filename, "w");
43 if( fsvg == NULL ) error("init_svg: unable to open SVG output file.");
44
45 /* write SVG header */
46 fprintf( fsvg,"<?xml version=\"1.0\" standalone=\"no\"?>\n");
47 fprintf( fsvg,"<!DOCTYPE svg PUBLIC \"-//W3C//DTD SVG 1.1//EN\"\n");
48 fprintf( fsvg," \"http://www.w3.org/Graphics/SVG/1.1/DTD/svg11.dtd\">\n");
49 fprintf( fsvg,"<svg width=\"%upx\" height=\"%upx\" ",xsize,ysize);
50 fprintf( fsvg,"version=\"1.1\"\n xmlns=\"http://www.w3.org/2000/svg\" ");
51 fprintf( fsvg,"xmlns:xlink=\"http://www.w3.org/1999/xlink\">\n");
52
53 return fsvg;
54}

76

79 {

94void write_svg_circ_arc( FILE *fsvg, Ring *cring )
95{
96 int fa, fs;
97 double ang_start, ang_end, C;
98 double x1, y1, x2, y2;
99
100 /* consider small arc, in trigonometric sense */
101 fa=0; fs=1;
102
103 /* compute angles delimiting the arc */
104 ang_start = atan2( cring->y1 - cring->cy, cring->x1 - cring->cx);
105 ang_end = atan2( cring->y2 - cring->cy, cring->x2 - cring->cx);
106
107/* if (almost) complete circle, write full circle, not an arc */
108 C = M_2__PI * cring->ax;
109 if( (cring->full) || ( (angle_diff( ang_start, ang_end )
110 < M_2__PI * SQRT2 / C)
111 && (angle_diff_signed( ang_start, ang_end)>0) ) )
112 {
113 fprintf( fsvg,"<ellipse cx=\"%f\" cy=\"%f\" rx=\"%f\" ry=\"%f\" stroke="
114 "\"red\" stroke-width=\"%d\" fill=\"none\" /> \n",
115 cring->cx, cring->cy, cring->ax, cring->bx, 1 );
116 }
117 else /* write an arc */
118 {
119 /* compute starting and ending points of the arc */
120 x1 = cring->ax * cos(ang_start) + cring->cx;
121 y1 = cring->ax * sin(ang_start) + cring->cy;
122 x2 = cring->ax * cos(ang_end ) + cring->cx;
123 y2 = cring->ax * sin(ang_end ) + cring->cy;
124 if( (double_equal(x1,x2) && double_equal(y1,y2)) || dist(x1,y1,x2,y2) < 2.0 )
125 {
126 fprintf( fsvg,"<ellipse cx=\"%f\" cy=\"%f\" rx=\"%f\" ry=\"%f\" stroke="
127 "\"red\" stroke-width=\"%d\" fill=\"none\" /> \n",
128 cring->cx, cring->cy, cring->ax, cring->bx, 1 );
129 return;
130 }
131 /* make sure delimiting angles are positive and ordered, to be able to
132 choose between small/big arc */
133 if( ang_start < 0 ) ang_start += M_2__PI;
134 if( ang_end < 0 ) ang_end += M_2__PI;
135
136 if( ang_end < ang_start ) ang_end += M_2__PI;
137
138 if( (ang_end - ang_start) > M_PI) fa = 1;
139 /* write starting and ending points, axes, orientation, fa, fs */
140 fprintf( fsvg,"<path d=\"M %f,%f A%f,%f %f %d,%d %f,%f\"",
141 x1, y1, cring->ax, cring->ax, 0.0, fa, fs, x2, y2 );
142 /* define style: color and width */
143 fprintf(fsvg," fill=\"none\" stroke =\"red\" stroke-width=\"%d\" />\n",1);
144 }
145}

152void write_svg_ell_arc( FILE *fsvg, Ring *ering )
153{
154 int fa, fs;
155 double ang_start, ang_end;
156 double x1, y1, x2, y2;
157
158 /* consider small arc, in trigonometric sense */
159 fa=0; fs=1;
160
161 if( ering->full ) /* if complete ellipse, write full ellipse, not an arc */
162 {
163 fprintf(fsvg,"<ellipse transform=\"translate(%f %f) rotate(%f)\" rx=\"%f\" "
164 "ry=\"%f\" stroke=\"red\" fill=\"none\" stroke-width=\"%d\" />\n",
165 ering->cx, ering->cy, (ering->theta)*180/M_PI, ering->ax, ering->bx, 1 );
166 }
167 else /* compute limits of the arc and write it */
168 {
169 /* compute starting point */
170 rosin_point( ering, ering->x1, ering->y1, &x1, &y1 );
171 /* compute ending point */
172 rosin_point( ering, ering->x2, ering->y2, &x2, &y2 );
173 if( (double_equal(x1,x2) && double_equal(y1,y2)) || dist(x1,y1,x2,y2) < 2.0 )
174 {
175 fprintf(fsvg,"<ellipse transform=\"translate(%f %f) rotate(%f)\" rx=\"%f\" "
176 "ry=\"%f\" stroke=\"red\" fill=\"none\" stroke-width=\"%d\" />\n",
177 ering->cx, ering->cy, (ering->theta)*180/M_PI, ering->ax, ering->bx, 1 );
178 return;
179 }
180 /* compute angles delimiting the arc */
181 ang_start = atan2( y1 - ering->cy, x1 - ering->cx );
182 ang_end = atan2( y2 - ering->cy, x2 - ering->cx );
183
184 /* make sure delimiting angles are positive and ordered, to be able to
185 choose between small/big arc */
186 if( ang_start < 0 ) ang_start += M_2__PI;
187 if( ang_end < 0 ) ang_end += M_2__PI;
188
189 if( ang_end < ang_start ) ang_end += M_2__PI;
190
191 if( (ang_end - ang_start ) > M_PI ) fa = 1;
192
193 /* write starting and ending points, axes, orientation, fa, fs */
194 fprintf( fsvg,"<path d=\"M %f,%f A%f,%f %f %d,%d %f,%f\"",
195 x1, y1, ering->ax, ering->bx, ering->theta*180/M_PI,
196 fa, fs, x2, y2);
197 /* define style: color and width */
198 fprintf(fsvg," fill=\"none\" stroke =\"red\" stroke-width=\"%d\" />\n",1);
199 }
