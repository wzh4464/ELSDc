219PImageDouble img_gradient_angle( PImageDouble in, double threshold )
220{
221 unsigned int xsize, ysize, adr, i, j;
222 double com1, com2, gx, gy, norm, norm2;
223 PImageDouble angles;
224
225 /* check parameters */
226 if( in == NULL || in->data == NULL || in->xsize <= 0 || in->ysize <= 0 )
227 error("img_gradient_angle: invalid input image");
228 if( threshold < 0.0 ) error("img_gradient_angle: threshold must be positive");
229
230 xsize = in->xsize;
231 ysize = in->ysize;
232
233 /* allocate output image */
234 angles = new_PImageDouble(xsize,ysize);
235
236 /* 'undefined' on the down and right boundaries */
237 for( i=0; i<ysize; i++) angles->data[ i * xsize + xsize - 1 ] = NOTDEF;
238 for( j=0; j<xsize; j++) angles->data[ (ysize-1) * xsize + j ] = NOTDEF;
239
240 /*** remaining part ***/
241 for( i=0; i<ysize-1; i++ )
242 for( j=0; j<xsize-1; j++ )
243 {
244 adr = i*xsize+j;
245 /*
246 Norm 2 computation using 2x2 pixel window:
247 A B
248 C D
249 and
250 com1 = D-A, com2 = B-C.
251 Then
252 gx = B+D - (A+C) horizontal difference
253 gy = C+D - (A+B) vertical difference
254 com1 and com2 are just to avoid 2 additions.
255 */
256 com1 = in->data[ adr + xsize + 1 ] - in->data[ adr ];
257 com2 = in->data[ adr + 1 ] - in->data[ adr + xsize ];
258 gx = com1 + com2;
259 gy = com1 - com2;
260 norm2 = gx*gx+gy*gy;
261 norm = sqrt( norm2 / 4.0 );
262
263 if( norm <= threshold ) /* norm too small, gradient not defined */
264 angles->data[ adr ] = NOTDEF;
265 else
266 angles->data[ adr ] = atan2( gy, gx );
267 }
268
269 return angles;
270}

277void img_gradient_sort( PImageDouble in, double threshold, CoordList **list_p,
278 void **mem_p, unsigned int n_bins, double max_grad,
279 PImageDouble *angles, PImageDouble *gradmag,
280 PImageDouble *gradx, PImageDouble *grady )
281{
282 unsigned int xsize, ysize, adr, ind, i, j;
283 double com1, com2, gx, gy, norm, norm2;
284
285 /* the rest of the variables are used for pseudo-ordering
286 the gradient magnitude values */
287 int list_count = 0;
288 CoordList *list;
289 CoordList **range_l_s; /* array of pointers to start of bin list */
290 CoordList **range_l_e; /* array of pointers to end of bin list */
291 CoordList *start;
292 CoordList *end;
293
294 /* check parameters */
295 if( (in == NULL) || (in->data == NULL) ||
296 (in->xsize) <= 0 || (in->ysize <= 0 ) )
297 error("img_gradient_sort: invalid input image.");
298 if( threshold < 0.0 )
299 error("img_gradient_sort: 'threshold' must be positive.");
300 if( list_p == NULL ) error("img_gradient_sort: NULL pointer 'list_p'.");
301 if( mem_p == NULL ) error("img_gradient_sort: NULL pointer 'mem_p'.");
302 if( n_bins <= 0 ) error("img_gradient_sort: 'n_bins' must be positive.");
303 if( max_grad <= 0.0 )
304 error("img_gradient_sort: 'max_grad' must be positive.");
305
306 xsize = in->xsize;
307 ysize = in->ysize;
308
309 /* allocate output angles image */
310 *angles = new_PImageDouble( in->xsize, in->ysize );
311 /* allocate image of gradient modulus and gradient vector components
312 on Ox and Oy */
313 *gradmag = new_PImageDouble( in->xsize, in->ysize );
314 *gradx = new_PImageDouble( in->xsize, in->ysize );
315 *grady = new_PImageDouble( in->xsize, in->ysize );
316
317 /* get memory for "ordered" coordinate list */
318 list = (CoordList *) calloc( (size_t) (xsize*ysize), sizeof(CoordList) );
319 *mem_p = (void *) list;
320 range_l_s = (CoordList **) calloc( n_bins, sizeof(CoordList *) );
321 range_l_e = (CoordList **) calloc( n_bins, sizeof(CoordList *) );
322 if( (list == NULL) || (range_l_s == NULL) || (range_l_e == NULL) )
323 error("ll_angle: not enough memory for sorted list.");
324
325 for( i=0; i<n_bins; i++ )
326 range_l_s[i] = range_l_e[i] = NULL;
327
328 /* 'undefined' on the down and right boundaries */
329 for( i=0; i<ysize; i++ ) (*angles)->data[ i * xsize + xsize - 1 ] = NOTDEF;
330 for( j=0; j<xsize; j++ ) (*angles)->data[ (ysize-1) * xsize + j ] = NOTDEF;
331
332 /*** remaining part ***/
333 for( i=0; i<ysize-1; i++ )
334 for( j=0; j<xsize-1; j++ )
335 {
336 adr = i * xsize + j;
337 /*
338 Norm 2 computation using 2x2 pixel window:
339 A B
340 C D
341 and
342 com1 = D-A, com2 = B-C.
343 Then
344 gx = B+D - (A+C) horizontal difference
345 gy = C+D - (A+B) vertical difference
346 com1 and com2 are just to avoid 2 additions.
347 */
348 com1 = in->data[ adr + xsize + 1 ] - in->data[ adr ];
349 com2 = in->data[ adr + 1 ] - in->data[ adr + xsize ];
350 gx = com1 + com2;
351 gy = com1 - com2;
352 norm2 = gx*gx + gy*gy;
353 norm = sqrt( norm2 / 4.0 );
354
355 (*gradmag)->data[adr] = norm;
356 (*gradx)->data [adr] = gx/2.0;
357 (*grady)->data [adr] = gy/2.0;
358 if( norm <= threshold ) /* norm too small, gradient not defined */
359 {
360 (*angles)->data[adr] = NOTDEF;
361 }
362 else
363 {
364 /* angle computation */
365 (*angles)->data[adr] = atan2( gy, gx );
366 /* store the point in the right bin according to its norm */
367 ind = (unsigned int) (norm * (double) n_bins / max_grad);
368 if( ind >= n_bins ) ind = n_bins - 1;
369 if( range_l_e[ind] == NULL)
370 range_l_s[ind] = range_l_e[ind] = list + list_count++;
371 else
372 {
373 range_l_e[ind]->next = list + list_count;
374 range_l_e[ind] = list + list_count++;
375 }
376 range_l_e[ind]->x = (int) j;
377 range_l_e[ind]->y = (int) i;
378 range_l_e[ind]->next = NULL;
379 }
380 }
381
382 /* Make the list of points "ordered" by norm value.
383 It starts by the larger bin, so the list starts by the
384 pixels with higher gradient value. */
385 for( i=n_bins-1; (i>0) && (range_l_s[i]==NULL); i-- ) {};
386 start = range_l_s[i];
387 end = range_l_e[i];
388 if( start != NULL )
389 for( i--; i>0 ; i-- )
390 if( range_l_s[i] != NULL )
391 {
392 end->next = range_l_s[i];
393 end = range_l_e[i];
394 }
395 *list_p = start;
396
397 /* free memory */
398 free( (void *) range_l_s );
399 free( (void *) range_l_e );
400}
