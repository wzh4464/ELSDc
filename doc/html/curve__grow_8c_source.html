49static void region_grow( PImageDouble angles, PImageInt used, Point *reg,
50 int start, int idx_buff, int *end, int label,
51 int label_start, double prec, double *reg_angle )
52{
53 double sumdx = 0.0, sumdy = 0.0;
54 double ang;
55 int xx, yy, i;
56 int adr;
57 /* check parameters */
58 if( (angles == NULL) || (angles->data == NULL) )
59 error("region_grow: invalid image 'angles'.");
60 if( (used == NULL) || (used->data == NULL) )
61 error("region_grow: invalid image 'used'.");
62 if( reg == NULL ) error("region_grow: invalid input region.");
63 /* if( start >= *end ) error("region_grow: invalid indexes.");*/
64 if( prec < 0.0 ) error("region_grow: 'prec' must be positive.");
65
66 /* Initialise *reg_angle with the value of the pixels
67 orientations in the buffer part of reg. */
68
69 for( i=start; i<idx_buff; i++ )
70 {
71 xx = reg[i].x; yy = reg[i].y;
72 adr = xx + yy*angles->xsize;
73 sumdx += cos( angles->data[adr] );
74 sumdy += sin( angles->data[adr] );
75 used->data[adr] = label;
76 }
77 *reg_angle = atan2( sumdy, sumdx );
78 /* try neighbors as new region points */
79 for( i=start; i<*end; i++ )
80 for( xx=reg[i].x-1; xx<=reg[i].x+1; xx++ )
81 for( yy=reg[i].y-1; yy<=reg[i].y+1; yy++ )
82 if( in_image( xx, yy, used->xsize, used->ysize ) )
83 {
84 adr = yy * angles->xsize + xx;
85 if( (used->data[adr] != USED) && (used->data[adr] != label) &&
86 (used->data[adr] < label_start) )
87 {
88 ang = angles->data[adr];
89 if( (ang != NOTDEF ) && (is_aligned( ang, *reg_angle, prec )) )
90 {
91 /* add point and mark as already visited in this call */
92 used->data[adr] = label;
93 reg[*end].x = xx;
94 reg[*end].y = yy;
95 ++(*end);
96 /* update region's angle */
97 sumdx += cos(ang);
98 sumdy += sin(ang);
99 *reg_angle = atan2( sumdy, sumdx );
100 }
101
102 }
103 }
104}

123static double get_theta( PImageDouble gradmag, Point *reg, int start, int end,
124 double cx, double cy, double reg_angle, double prec )
125{
126 double lambda1, lambda2, tmp, theta, weight, sum;
127 double Ixx = 0.0;
128 double Iyy = 0.0;
129 double Ixy = 0.0;
130 int i;
131 /* check parameters */
132 if( reg == NULL ) error("get_theta: invalid input region.");
133 if( (gradmag == NULL) || (gradmag->data == NULL) )
134 error("get_theta: invalid image 'gradmag'.");
135 if( start >= end ) error("get_theta: invalid indexes.");
136 if( prec < 0.0 ) error("get_theta: 'prec' must be positive.");
137
138 /*----------- theta ---------------------------------------------------*/
139 /*
140 Region inertia matrix A:
141 Ixx Ixy
142 Ixy Iyy
143 where
144 Ixx = \sum_i y_i^2
145 Iyy = \sum_i x_i^2
146 Ixy = -\sum_i x_i y_i
147
148 lambda1 and lambda2 are the eigenvalues, with lambda1 >= lambda2.
149 They are found by solving the characteristic polynomial
150 det(\lambda I - A) = 0.
151
152 To get the line segment direction we want to get the eigenvector of
153 the smaller eigenvalue. We have to solve a,b in:
154 a.Ixx + b.Ixy = a.lambda2
155 a.Ixy + b.Iyy = b.lambda2
156 We want the angle theta = atan(b/a). I can be computed with
157 any of the two equations:
158 theta = atan( (lambda2-Ixx) / Ixy )
159 or
160 theta = atan( Ixy / (lambda2-Iyy) )
161
162 When |Ixx| > |Iyy| we use the first, otherwise the second
163 (just to get better numeric precision).
164 */
165 sum = 0.0;
166 for( i=start; i<end; i++ )
167 {
168 weight = gradmag->data[ reg[i].x + reg[i].y * gradmag->xsize ];
169 Ixx += ( (double) reg[i].y - cy ) * ( (double) reg[i].y - cy ) * weight;
170 Iyy += ( (double) reg[i].x - cx ) * ( (double) reg[i].x - cx ) * weight;
171 Ixy -= ( (double) reg[i].x - cx ) * ( (double) reg[i].y - cy ) * weight;
172 sum += weight;
173 }
174 if( sum <= 0.0 ) error("get_theta: weights sum less or equal to zero.");
175 Ixx /= sum;
176 Iyy /= sum;
177 Ixy /= sum;
178 lambda1 = ( Ixx + Iyy + sqrt( (Ixx-Iyy)*(Ixx-Iyy) + 4.0*Ixy*Ixy ) ) / 2.0;
179 lambda2 = ( Ixx + Iyy - sqrt( (Ixx-Iyy)*(Ixx-Iyy) + 4.0*Ixy*Ixy ) ) / 2.0;
180 if( fabs(lambda1) < fabs(lambda2) )
181 {
182 fprintf(stderr,"Ixx %g Iyy %g Ixy %g lamb1 %g lamb2 %g - lamb1 < lamb2\n",
183 Ixx, Iyy, Ixy, lambda1, lambda2 );
184 tmp = lambda1;
185 lambda1 = lambda2;
186 lambda2 = tmp;
187 }
188
189 if( fabs(Ixx) > fabs(Iyy) )
190 theta = atan2( lambda2-Ixx, Ixy );
191 else
192 theta = atan2( Ixy, lambda2-Iyy );
193
194 /* The previous procedure doesn't care about orientation,
195 so it could be wrong by 180 degrees. Here is corrected if necessary. */
196 if( angle_diff( theta-M_1_2_PI, reg_angle ) > prec ) theta += M_PI;
197
198 /* If the estimated orientation is still very different wrt reg_angle
199 (region is nearly square), return the orthogonal to original reg_angle. */
200 if( angle_diff( theta-M_1_2_PI, reg_angle ) > prec )
201 theta = reg_angle + M_1_2_PI;
202
203 while( theta > M_PI ) theta -= M_2__PI;
204 while( theta < -M_PI ) theta += M_2__PI;
205 return theta;
206}

209/*----------------------------------------------------------------------------*/
214void region2rect( PImageDouble gradmag, Point *reg, int start, int end,
215 double reg_angle, Rectangle *rec, double prec )
216{
217 double cx, cy, dx, dy, l, w, theta;
218 double weight,sum,l_min,l_max,w_min,w_max;
219 int i, adr;
220 /* Check parameters */
221 if( (gradmag == NULL) || (gradmag->data == NULL) )
222 error("region2rect: invalid image 'gradmag'.");
223 if( reg == NULL ) error("region2rect: invalid input region.");
224 if( start >= end ) error("region2rect: invalid indexes.");
225 if( rec == NULL ) error("region2rect: invalid 'rec'.");
226
227 /* Center */
228 cx = cy = sum = 0.0;
229 for( i=start; i<end; i++ )
230 {
231 adr = reg[i].x + reg[i].y * gradmag->xsize;
232 weight = gradmag->data[adr];
233 cx += (double) reg[i].x * weight;
234 cy += (double) reg[i].y * weight;
235 sum += weight;
236 }
237
238 cx /= sum;
239 cy /= sum;
240
241 /* Orientation */
242 theta = get_theta( gradmag, reg, start, end, cx, cy, reg_angle, prec );
243
244 /* Length and width */
245 dx = cos(theta);
246 dy = sin(theta);
247 l_min = l_max = w_min = w_max = 0.0;
248 for( i=start; i<end; i++ )
249 {
250 l = ( (double) reg[i].x - cx) * dx + ( (double) reg[i].y - cy) * dy;
251 w = -( (double) reg[i].x - cx) * dy + ( (double) reg[i].y - cy) * dx;
252
253 if( l > l_max ) l_max = l;
254 if( l < l_min ) l_min = l;
255 if( w > w_max ) w_max = w;
256 if( w < w_min ) w_min = w;
257 }
258
259 /* Store values */
260 rec->x1 = cx + l_min * dx;
261 rec->y1 = cy + l_min * dy;
262 rec->x2 = cx + l_max * dx;
263 rec->y2 = cy + l_max * dy;
264 rec->width = w_max - w_min;
265 rec->wmin = w_min;
266 rec->wmax = w_max;
267 rec->cx = cx;
268 rec->cy = cy;
269 rec->theta = theta;
270 rec->dx = dx;
271 rec->dy = dy;
272 rec->prec = prec;
273 rec->len = dist( rec->x1, rec->y1, rec->x2, rec->y2 );
274 /* Correct if rectangle too thin */
275 if( rec->width < 1.0 )
276 {
277 rec->width = 1.0; rec->wmin = -0.5; rec->wmax = 0.5;
278 }
279}

287static int reduce_region_radius( PImageDouble gradmag, PImageDouble angles,
288 PImageInt used, Point *reg, int start,
289 int *end, double reg_angle, Rectangle *rec,
290 int x, int y, double density_th, double prec )
291{
292 double density, rad1, rad2, rad;
293 int i;
294 /* Check parameters */
295 if( (gradmag == NULL) || (gradmag->data == NULL) )
296 error("reduce_region_radius: invalid image 'gradmag'.");
297 if( (angles == NULL) || (angles->data == NULL) )
298 error("reduce_region_radius: invalid image 'angles'.");
299 if( (used == NULL) || (used->data == NULL) )
300 error("reduce_region_radius: invalid image 'used'.");
301 if( reg == NULL ) error("reduce_region_radius: invalid input region.");
302 if( start >= *end ) error("reduce_region_radius: invalid indexes.");
303 if( rec == NULL ) error("reduce_region_radius: invalid 'rec'.");
304 if( prec < 0.0 ) error("reduce_region_radius: 'prec' must be positive.");
305
306 /* Compute region points density */
307 density = (double) (*end - start) /
308 (dist(rec->x1,rec->y1,rec->x2,rec->y2) * rec->width);
309
310 if( density >= density_th ) return TRUE;
311
312 /* Compute region radius */
313 rad1 = dist( (double)x, (double)y, rec->x1, rec->y1 );
314 rad2 = dist( (double)x, (double)y, rec->x2, rec->y2 );
315 rad = rad1 > rad2 ? rad1 : rad2;
316
317 while( density < density_th )
318 {
319 rad *= 0.75;
320
321 /* Remove points from the region and update 'used' map */
322 for( i=start; i<*end; i++ )
323 if( dist( x, y, (double) reg[i].x, (double) reg[i].y ) > rad )
324 {
325 /* Point not kept, mark it as NOTUSED */
326 used->data[ reg[i].x + reg[i].y * used->xsize ] = NOTUSED;
327 /* Remove point from the region */
328 reg[i].x = reg[*end-1].x; /* if i==*end-1 copy itself */
329 reg[i].y = reg[*end-1].y;
330 --(*end);
331 --i; /* to avoid skipping one point */
332 }
333 /* Reject if the region is too small.
334 2 is the minimal region size for 'region2rect' to work. */
335
336 if( (*end - start) < 2 ) return FALSE;
337
338 /* Re-compute rectangle */
339 region2rect( gradmag, reg, start, *end, reg_angle, rec, prec );
340
341 /* Re-compute region points density */
342 density = (double) ( *end - start) /
343 ( dist( rec->x1, rec->y1, rec->x2, rec->y2 ) * rec->width );
344 }
345 return TRUE;
346}

358static int refine( PImageDouble gradmag, PImageDouble angles, PImageInt used,
359 Point *reg, int start, int idx_buff, int *end, int *label,
360 int label_start, Rectangle *rec, double density_th,
361 double prec )
362{
363 double angle, ang_d, mean_angle, tau, density, ang_c, sum, s_sum, reg_angle;
364 double gradmax;
365 int i, n, xx, yy;
366 int xc, yc;
367
368 /* Check parameters */
369 if( (gradmag == NULL) || (gradmag->data == NULL) )
370 error("refine: invalid image 'gradmag'.");
371 if( (angles == NULL) || (angles->data == NULL) )
372 error("refine: invalid image 'angles'.");
373 if( (used == NULL) || (used->data == NULL) )
374 error("refine: invalid image 'used'.");
375 if( reg == NULL ) error("refine: invalid input region.");
376 if( start >= *end ) error("refine: invalid indexes.");
377 if( *label <= 1 ) error("refine: forbidden label value.");
378 if( prec < 0.0 ) error("refine: 'prec' must be positive.");
379 if( rec == NULL ) error("refine: invalid pointer 'rec'.");
380
381 /* Compute region points density */
382 density = (double) (*end - start) /
383 ( dist(rec->x1,rec->y1,rec->x2,rec->y2) * rec->width );
384
385 if( density >= density_th ) return TRUE;
386
387 /*------ First try: reduce angle tolerance ------*/
388
389 /* Compute the new mean angle and tolerance */
390 gradmax = 0.0;
391 xc = reg[start].x;
392 yc = reg[start].y;
393
394 for( i=start; i<idx_buff; i++ )
395 {
396 xx = reg[i].x; yy = reg[i].y;
397 if( gradmag->data[yy*gradmag->xsize+xx] > gradmax )
398 {
399 xc = xx;
400 yc = yy;
401 gradmax = gradmag->data[yy*gradmag->xsize+xx];
402 }
403 }
404 ang_c = angles->data[ xc + yc * angles->xsize ];
405 sum = s_sum = 0.0;
406 n = 0;
407 for( i=start; i<*end; i++ )
408 {
409 used->data[ reg[i].x + reg[i].y * used->xsize ] = NOTUSED;
410 if( dist( (double)xc,(double)yc,(double)reg[i].x,(double)reg[i].y )
411 < rec->width )
412 {
413 angle = angles->data[ reg[i].x + reg[i].y * angles->xsize ];
414 ang_d = angle_diff_signed(angle,ang_c);
415 sum += ang_d;
416 s_sum += ang_d *ang_d;
417 ++n;
418 }
419 }
420 mean_angle = sum / (double) n;
421 /* tau = 2 * standard deviation */
422 tau = max(2.0 * sqrt( (s_sum - 2.0 * mean_angle * sum) / (double) n
423 + mean_angle * mean_angle ),0.2);
424 tau = min(tau,prec);
425 /*tau = 2.0 * sqrt( (s_sum - 2.0 * mean_angle * sum) / (double) n
426 + mean_angle*mean_angle );*/ /* 2 * standard deviation */
427
428
429 /* Find a new region from the same starting point and new angle tolerance */
430 *end = idx_buff;
431 (*label)++;
432 region_grow( angles, used, reg, start, idx_buff, end, *label, label_start,
433 tau, &reg_angle );
434
435 /* If the region is too small, reject */
436 if( *end - start <= 2 ) return FALSE;
437
438 /* Re-compute rectangle */
439 region2rect( gradmag, reg, start, *end, reg_angle, rec, prec );
440
441 /* Re-compute region points density */
442 density = (double) (*end - start) /
443 ( dist(rec->x1,rec->y1,rec->x2,rec->y2) * rec->width);
444
445 /*------ Second try: reduce region radius ------*/
446 if( density < density_th )
447 return reduce_region_radius( gradmag, angles, used, reg, start, end,
448 reg_angle, rec, xc, yc, density_th, prec );
449 return TRUE;
450}

463static void px_seed( PImageDouble angles, PImageInt used, Rectangle *rec,
464 Point *reg, int start, int *idx_buff, int end,
465 double *pext, double prec, int label, int label_start )
466{
467 int xx, yy, i;
468 int adr;
469 int idx_buff0 = *idx_buff;
470 double width;
471 double l1, l2, l;
472
473 /* Check parameters */
474 if( (angles == NULL) || (angles->data == NULL) )
475 error("px_seed: invalid input 'angles' image.");
476 if( (used == NULL) || (used->data == NULL) )
477 error("px_seed: invalid input 'used' image.");
478 if( rec == NULL ) error("px_seed: invalid input rectangle.");
479 if( reg == NULL ) error("px_seed: invalid input region.");
480 if( end <= start ) error("px_seed: invalid region size.");
481
482 width = rec->width/2.0;
483
484 /* Distance between rectangle centre and extreme point; a new point can be
485 a seed only if it is at a distance farther than this */
486 l1 = fabs( (pext[0] - rec->cx) * rec->dx + (pext[1] - rec->cy) * rec->dy);
487
488 /* Scan previous region to find valid points that are close to the end of
489 the rectangle; the new seed pixels will be among the neighbours of these
490 points. */
491
492 for( i=start; i<end; i++ )
493 {
494 l = fabs( ( (double)reg[i].x - pext[0] ) * rec->dx +
495 ( (double)reg[i].y - pext[1] ) * rec->dy);
496
497 if( l<width )
498 for( xx=reg[i].x-1; xx<=reg[i].x+1; xx++ )
499 for( yy=reg[i].y-1; yy<=reg[i].y+1; yy++ )
500 {
501 /* If point in image */
502 if( in_image( xx, yy, used->xsize, used->ysize ) )
503 {
504 adr = yy * used->xsize + xx;
505 /* If point not already used or visited or notdef */
506 if( (used->data[adr] != USED) && (used->data[adr] != label ) &&
507 (angles->data[adr] != NOTDEF) && used->data[adr] < label_start )
508 {
509 l2 = fabs( (xx - rec->cx) * rec->dx +
510 (yy - rec->cy) * rec->dy );
511 /* If point's orientation is similar and the point is at
512 an acceptable distance */
513 if( (angle_diff( angles->data[adr],
514 rec->theta-M_1_2_PI ) < 3 * prec ) && (l2 >= l1-1) )
515 {
516 reg[ *idx_buff ].x = xx;
517 reg[ *idx_buff ].y = yy;
518 ++(*idx_buff);
519 used->data[adr] = label;
520 }
521 }
522 }
523 }
524 }
525
526 /* If none of the visited pixels satisfy the conditions, take as seed
527 the end of the rectangle, if it is inside the image */
528 if( idx_buff0 == *idx_buff )
529 {
530 xx = floor(pext[0]); yy = floor(pext[1]);
531 if( in_image( xx, yy, used->xsize, used->ysize ) )
532 {
533 reg[ *idx_buff ].x = xx;
534 reg[ *idx_buff ].y = yy;
535 ++(*idx_buff);
536 }
537 }
538}

541/*----------------------------------------------------------------------------*/
547static void subcurve( PImageDouble angles, PImageDouble gradmag, PImageInt used,
548 PolyRect *poly, Point *reg, int start, int idx_buff,
549 int *end, double *pext, double *spir, double density_th,
550 double prec,int *sgn, int *label, int label_start, int w )
551{
552 int convex = 1;
553 double reg_angle, difang, ang_prev;
554 double lenrec, lenth;
555 int buf_tmp = 0;
556 Rectangle rec;
557
558 /* Check parameters */
559 if( (angles == NULL) || (angles->data == NULL) )
560 error("subcurve: invalid input 'angles' image.");
561 if( (gradmag == NULL) || (gradmag->data == NULL) )
562 error("subcurve: invalid input 'gradmag' image.");
563 if( (used == NULL) || (used->data == NULL) )
564 error("subcurve: invalid input 'used' image.");
565 if( reg == NULL ) error("subcurve: invalid input region.");
566 if( *end <= start ) error("subcurve: invalid region size.");
567 if( *label <= 1 ) error("subcurve: forbidden label value.");
568
569
570 /* Get length and angle of initial rectangle (stored in polygon on first
571 position) */
572 lenth = poly->rectlist[0].len;
573 ang_prev = poly->rectlist[0].theta;
574
575 /* Add new rectangles while the conditions of smoothness and convexity
576 are satisfied */
577 while( convex )
578 {
579 (*label)++;
580 start = *end;
581 *end = idx_buff;
582 buf_tmp = *end - start;
583 /* Start region grow at the end of the previous rectangle */
584 region_grow( angles, used, reg, start, idx_buff, end, *label, label_start,
585 prec, &reg_angle );
586 if( *end - start > buf_tmp )
587 {
588 /* Estimate and refine rectangle */
589 region2rect( gradmag, reg, start, *end, reg_angle, &rec, prec );
590 (*label)++;
591 if( !refine( gradmag, angles, used, reg, start, idx_buff, end,
592 label, label_start, &rec, density_th, prec) )
593 {
594 /* If operation not successful, clean last visited points and
595 return */
596 mark_img_pts( used, reg, start, *end, NOTUSED );
597 *end = start;
598 break;
599 }
600 if( *end - start <= buf_tmp )
601 {
602 *end = start;
603 break;
604 }
605
606 /* Convexity rule imposes that consecutive rectangles turn in the same
607 direction, hence must have the same sign for angle difference
608 between consecutive rectangles' orientation. If set, verify sign
609 before adding a new rectangle. If not set (at first call),
610 then set sign. */
611 if( *sgn == 0 )
612 *sgn = sign( angle_diff_signed( rec.theta, ang_prev ) );
613
614 /* Get angle difference between current rectangle and previous
615 rectangle */
616 difang = angle_diff_signed( rec.theta, ang_prev );
617
618 /* If consecutive rectangles don't have comparable lengths
619 (condition for smoothness): clean and return;
620 Special case: last rectangle that joins the ends of a full
621 chain can be shorter than previous. */
622 lenrec = rec.len;
623 if( lenrec > lenth*3.0 ||
624 (lenrec < lenth/3.0 && (*spir + fabs(difang)) <= 0.9*M_2__PI) )
625 {
626 mark_img_pts( used, reg, start, *end, NOTUSED );
627 *end = start;
628 break;
629 }
630 lenth = lenrec;
631
632
633 /* Convexity & non-spiral check */
634 if( (sign(difang) == *sgn) && (fabs(difang) < (M_PI/3.0) ) &&
635 ( (*spir = *spir + fabs(difang)) <= M_2__PI) )
636 {
637 /* If all conditions are met, find next seed point,
638 and add rectangle to polygon */
639 (*label)++;
640 idx_buff = *end;
641 if( w==0 ) {pext[0] = rec.x1; pext[1] = rec.y1;}
642 else {pext[0] = rec.x2; pext[1] = rec.y2;}
643 px_seed( angles, used, &rec, reg, start, &idx_buff, *end,
644 pext, prec, *label, label_start );
645
646 /* Add rectangle to polygon */
647 poly->dim++;
648 add_rect_to_polyrect( poly, &rec );
649 ang_prev = rec.theta;
650 }
651 else /* Not convex */
652 {
653 convex = 0;
654 /* Subtract the contribution of the last rectangle and clean */
655 mark_img_pts( used, reg, start, *end, NOTUSED );
656 *end = start;
657 }
658 } /* !if( *end - start > 1 ) */
659 else
660 {
661 convex = 0;
662 *end = start;
663 }
664 } /* !while(convex) */
665}

668/*----------------------------------------------------------------------------*/
672int curve_grow( PImageDouble gradmag, PImageDouble angles,
673 PImageInt used, Point *reg, int *reg_size,
674 double density_th, double prec, PolyRect *poly,
675 int *label, double *pext1, double *pext2, double *spir )
676{
677 int sgn = 0;
678 Rectangle rec;
679 int size_first_reg = 0;
680 int label_start;
681 int start, idx_buff;
682 double reg_angle;
683
684 /* Check parameters */
685 if( (gradmag == NULL) || (gradmag->data == NULL) )
686 error("curve_grow: invalid input 'gradmag' image.");

754 subcurve( angles, gradmag, used, poly, reg, start, idx_buff, reg_size,
755 pext2, spir, density_th, prec, &sgn, label, label_start, 0 );
