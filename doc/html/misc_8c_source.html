48#define RELATIVE_ERROR_FACTOR 100.0
49int double_equal( double a, double b )
50{
51 double abs_diff, aa, bb, abs_max;
52 if( a == b ) return TRUE;
53
54 abs_diff = fabs( a-b );
55 aa = fabs(a);
56 bb = fabs(b);
57 abs_max = aa > bb ? aa : bb;
58
59 /* DBL_MIN is the smallest normalized number, thus, the smallest
60 number whose relative error is bounded by DBL_EPSILON. For
61 smaller numbers, the same quantization steps as for DBL_MIN
62 are used. Then, for smaller numbers, a meaningful "relative"
63 error should be computed by dividing the difference by DBL_MIN. */
64 if( abs_max < DBL_MIN ) abs_max = DBL_MIN;
65
66 /* equal if relative error <= factor x eps */
67 return ( abs_diff / abs_max ) <= (RELATIVE_ERROR_FACTOR * DBL_EPSILON);
68}

155{
156 double m = DBL_MAX;
157 int i;
158
159 /* check parameters */
160 if( v == NULL ) error("min_array: null input array.");
161 if( sz <= 0 ) error("min_array: size must be strictly positive.");
162
163 for( i=0; i<sz; i++ )
164 if( v[i] < m )
165 m = v[i];

228int is_aligned( double a, double theta, double prec )
229{
230 /* check parameters */
231 if( prec < 0.0 ) error("is_similar: 'prec' must be positive.");
232
233 /* if angle not defined, it is considered NON-similar */
234 if( a == NOTDEF ) return FALSE; /* there is no need to call the function
235 'double_equal' here because there is
236 no risk of problems related to the
237 comparison doubles, we are only
238 interested in the exact NOTDEF value */
239
240 /* it is assumed that 'theta' and 'a' are in the range [-pi,pi] */
241 theta -= a;
242 if( theta < 0.0 ) theta = -theta;
243 if( theta > M_3_2_PI )
244 {
245 theta -= M_2__PI;
246 if( theta < 0.0 ) theta = -theta;
247 }
248
249 return theta <= prec;
250}

256double norm_angle_diff( double a, double b )
257{
258 /* check if valid angles */
259 if( (a == NOTDEF) || (b == NOTDEF) ) return 1.0;
260
261 /* normalize a */
262 while( a <= -M_PI ) a += M_2__PI;
263 while( a > M_PI ) a -= M_2__PI;
264
265 /* normalize b */
266 while( b <= -M_PI ) b += M_2__PI;
267 while( b > M_PI ) b -= M_2__PI;
268
269 /* compute difference and return normalised value */
270 a -= b;
271 while( a <= -M_PI ) a += M_2__PI;
272 while( a > M_PI ) a -= M_2__PI;
273
274 return fabs(a) / M_PI;
275}
