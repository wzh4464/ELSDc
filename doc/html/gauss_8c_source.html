50static PGaussFilter new_PGaussFilter( unsigned int dim )
51{
52 PGaussFilter fil;
53
54 /* check parameters */
55 if( dim <= 0 ) error("new_PGaussFilter: invalid filter size");
56
57 fil = (PGaussFilter) malloc( sizeof( struct GaussFilter ) );
58 if( fil == NULL ) error("new_PGaussFilter: not enough memory");
59 fil->values = (double *) calloc( dim, sizeof(double) );
60 if( fil->values == NULL ) error("new_PGaussFilter: not enough memory");
61
62 fil->dim = dim;
63 fil->mean = 0.0;
64
65 return fil;
66}

77static void gaussian_kernel( PGaussFilter kernel, double sigma, double mean )
78{
79 double sum = 0.0;
80 double val;
81 int i;
82
83 /* check parameters */
84 if( (kernel == NULL) || (kernel->values == NULL) )
85 error("gaussian_kernel: invalid struct 'kernel'.");
86 if( sigma <= 0.0 ) error("gaussian_kernel: 'sigma' must be positive.");
87
88 /* compute gaussian kernel */
89 kernel->mean = mean;
90 kernel->sigma = sigma;
91
92 for( i=0; i<kernel->dim; i++ )
93 {
94 val = ( (double) i - mean ) / sigma;
95 kernel->values[i] = exp( -0.5 * val * val );
96 sum += kernel->values[i];
97 }
98
99 /* normalization */
100 if( sum >= 0.0 )
101 for( i=0; i<kernel->dim; i++ )
102 kernel->values[i] /= sum;
103}

115PImageDouble gaussian_sampler( PImageDouble in, double scale,
116 double sigma_scale )
117{
118 PImageDouble aux;
119 PImageDouble out;
120 PGaussFilter kernel;
121 unsigned int N, M, h, n, x, y;
122 int i;
123 int xc, yc, j, double_x_size, double_y_size;
124 double sigma, xx, yy, sum, prec;
125
126 /* check parameters */
127 if( (in == NULL) || (in->data == NULL) || (in->xsize <= 0) ||
128 (in->ysize <= 0) )
129 error("gaussian_sampler: invalid image.");
130 if( scale <= 0.0 ) error("gaussian_sampler: 'scale' must be positive.");
131 if( sigma_scale <= 0.0 )
132 error("gaussian_sampler: 'sigma_scale' must be positive.");
133
134 /* get memory for images */
135 N = (unsigned int) floor( in->xsize * scale );
136 M = (unsigned int) floor( in->ysize * scale );
137 aux = new_PImageDouble( N, in->ysize );
138 out = new_PImageDouble( N, M );
139
140 /* sigma, kernel size and memory for the kernel */
141 sigma = scale < 1.0 ? sigma_scale / scale : sigma_scale;
142
143 /*
144 The size of the kernel is selected to guarantee that the
145 the first discarded term is at least 10^prec times smaller
146 than the central value. For that, h should be larger than x, with
147 e^(-x^2/2sigma^2) = 1/10^prec.
148 Then,
149 x = sigma * sqrt( 2 * prec * ln(10) ).
150 */
151 prec = 3.0;
152 h = (unsigned int) ceil( sigma * sqrt( 2.0 * prec * log(10.0) ) );
153 n = 1 + 2*h; /* kernel size */
154 kernel = new_PGaussFilter(n);
155
156 /* auxiliary double image size variables */
157 double_x_size = (int) (2 * in->xsize);
158 double_y_size = (int) (2 * in->ysize);
159
160 /* First subsampling: x axis */
161 for( x=0; x<aux->xsize; x++ )
162 {
163 /*
164 x is the coordinate in the new image.
165 xx is the corresponding x-value in the original size image.
166 xc is the integer value, the pixel coordinate of xx.
167 */
168 xx = (double) x / scale;
169 /* coordinate (0.0,0.0) is in the center of pixel (0,0),
170 so the pixel with xc=0 get the values of xx from -0.5 to 0.5 */
171 xc = (int) floor( xx + 0.5 );
172 gaussian_kernel( kernel, sigma, (double) h + xx - (double) xc );
173 /* the kernel must be computed for each x because the fine
174 offset xx-xc is different in each case */
175
176 for( y=0; y<aux->ysize; y++ )
177 {
178 sum = 0.0;
179 for( i=0; i<kernel->dim; i++ )
180 {
181 j = xc - h + i;
182
183 /* symmetry boundary condition */
184 while( j < 0 ) j += double_x_size;
185 while( j >= double_x_size ) j -= double_x_size;
186 if( j >= (int) in->xsize ) j = double_x_size-1-j;
187
188 sum += in->data[ j + y * in->xsize ] * kernel->values[i];
189 }
190 aux->data[ x + y * aux->xsize ] = sum;
191 }
192 }
193
194 /* Second subsampling: y axis */
195 for( y=0; y<out->ysize; y++ )
196 {
197 /*
198 y is the coordinate in the new image.
199 yy is the corresponding x-value in the original size image.
200 yc is the integer value, the pixel coordinate of xx.
201 */
202 yy = (double) y / scale;
203 /* coordinate (0.0,0.0) is in the center of pixel (0,0),
204 so the pixel with yc=0 get the values of yy from -0.5 to 0.5 */
205 yc = (int) floor( yy + 0.5 );
206 gaussian_kernel( kernel, sigma, (double) h + yy - (double) yc );
207 /* the kernel must be computed for each y because the fine
208 offset yy-yc is different in each case */
209
210 for( x=0; x<out->xsize; x++ )
211 {
212 sum = 0.0;
213 for( i=0; i<kernel->dim; i++ )
214 {
215 j = yc - h + i;
216
217 /* symmetry boundary condition */
218 while( j < 0 ) j += double_y_size;
219 while( j >= double_y_size ) j -= double_y_size;
220 if( j >= (int) in->ysize ) j = double_y_size-1-j;
221
222 sum += aux->data[ x + j * aux->xsize ] * kernel->values[i];
223 }
224 out->data[ x + y * out->xsize ] = sum;
225 }
226 }
227
228 /* free memory */
229 free_PGaussFilter(kernel);
230 free_PImageDouble(aux);
231 return out;
232}
