41static double inter_low( double x, double x1, double y1, double x2, double y2 )
42{
43 if( (x1 > x2) || (x < x1) || (x > x2) )
44 {
45 fprintf( stderr,"inter_low: x %g x1 %g x2 %g.\n", x, x1, x2 );
46 error("inter_low: something went wrong!");
47 }
48 if( (double_equal( x1, x2 )) && (y1<y2) ) return y1;
49 if( (double_equal( x1, x2 )) && (y1>y2) ) return y2;
50 return y1 + (x-x1) * (y2-y1) / (x2-x1);
51}

93/*----------------------------------------------------------------------------*/
96void inc_RectIter( RectIter *i )
97{
98 /* check parameters */
99 if( i == NULL ) error("inc_RectIter: NULL iterator.");
100
101 if( (double) (i->x) <= i->vx[2] ) i->y++;
102 while( ((double) (i->y) > i->ye) && ((double) (i->x) <= i->vx[2]) )
103 {
104 /* new x */
105 i->x++;
106
107 if( (double) (i->x) > i->vx[2] ) return; /* end of iteration */
108
109 /* update lower y limit for the line */
110 if( (double) i->x < i->vx[3] )
111 i->ys = inter_low((double)i->x,i->vx[0],i->vy[0],i->vx[3],i->vy[3]);
112 else
113 i->ys = inter_low((double)i->x,i->vx[3],i->vy[3],i->vx[2],i->vy[2]);
114
115 /* update upper y limit for the line */
116 if( (double)i->x < i->vx[1] )
117 i->ye = inter_hi((double)i->x,i->vx[0],i->vy[0],i->vx[1],i->vy[1]);
118 else
119 i->ye = inter_hi((double)i->x,i->vx[1],i->vy[1],i->vx[2],i->vy[2]);
120
121 /* new y */
122 i->y = (int) ceil(i->ys);
123 }
124}

127/*----------------------------------------------------------------------------*/
130RectIter *ini_RectIter( Rectangle *r )
131{
132 double vx[4], vy[4];
133 int n, offset;
134 RectIter *i;
135
136 /* check parameters */
137 if( r == NULL ) error("ini_RectIter: invalid input rectangle.");
138
139 i = (RectIter *) malloc( sizeof(RectIter) );
140 if( i == NULL ) error("ini_RectIter: not enough memory.");
141
142 vx[0] = r->x1 - r->dy * r->wmax;
143 vy[0] = r->y1 + r->dx * r->wmax;
144 vx[1] = r->x2 - r->dy * r->wmax;
145 vy[1] = r->y2 + r->dx * r->wmax;
146 vx[2] = r->x2 - r->dy * r->wmin;
147 vy[2] = r->y2 + r->dx * r->wmin;
148 vx[3] = r->x1 - r->dy * r->wmin;
149 vy[3] = r->y1 + r->dx * r->wmin;
150
151 if ( (r->x1 < r->x2) && (r->y1 <= r->y2) ) offset = 0;
152 else if( (r->x1 >= r->x2) && (r->y1 < r->y2) ) offset = 1;
153 else if( (r->x1 > r->x2) && (r->y1 >= r->y2) ) offset = 2;
154 else offset = 3;
155
156 for( n=0; n<4; n++ )
157 {
158 i->vx[n] = vx[ (offset+n) % 4 ];
159 i->vy[n] = vy[ (offset+n) % 4 ];
160 }
161
162 /* starting point */
163 i->x = (int) ceil(i->vx[0]) - 1;
164 i->y = (int) ceil(i->vy[0]);
165 i->ys = i->ye = -DBL_MAX;
166
167 /* advance to the first point */
168 inc_RectIter(i);
169
170 return i;
171}
