41int main( int argc, char **argv )
42{
43 /* check arguments */
44 if( argc < 2 ) error("usage: ./elsdc image_name.pgm");
45
46 PImageDouble in; /* input image */
47 PImageInt out; /* output image having the same size as 'in'; the pixels
48 supporting a certain geometric primitive are marked
49 with the same label */
50
51 int ell_count = 0; /* number of detected ellipses */
52 int *ell_labels=NULL;/* the pixels supporting a certain ellipse are marked
53 with the same unique label */
54 Ring *ell_out = NULL;/* array containing the parameters of the detected
55 ellipses; correlated with ell_labels, i.e. the i-th
56 element of ell_labels is the label of the pixels
57 supporting the ellipse defined by the parameters
58 ell[i] */
59
60 int poly_count = 0; /* number of detected polygons */
61 int *poly_labels=NULL;/* the pixels supporting a certain polygon are marked
62 with the same unique label */
63 Polygon *poly_out=NULL;/* array containing the parameters of the detected
64 polygons; correlated with poly_labels, i.e. the i-th
65 element of ell_labels is the label of the pixels
66 supporting the polygon defined by the parameters
67 poly[i] */
68
69 FILE *ell_ascii; /* output file with the parameters of the detected
70 ellipses -- ASCII format */
71 FILE *poly_ascii; /* output file with the parameters of the detected
72 polygons -- ASCII format */
73 FILE *fsvg; /* output file with the detected ellipses and polygons
74 in vectorial form */
75 int i,j;
76
77 /* read input image; must be PGM form */
78 in = read_pgm_image_double( argv[1] );
79 int xsize = in->xsize, ysize = in->ysize;
80
81 /* create and initialize with 0 output label image */
82 out = new_PImageInt_ini( in->xsize, in->ysize, 0 );
83
84 /* call detection procedure */
85 ELSDc( in, &ell_count, &ell_out, &ell_labels, &poly_count, &poly_out,
86 &poly_labels, out );
87
88 /* write results in ASCII */
89 /* Ellipse file: each line contains 1 integer and 11 doubles in the form
90 ell_label x1 y1 x2 y2 cx cy ax bx theta ang_start ang_end
91 where:
92 ell_label -- the label of the pixels supporting this ellipse in
93 'out' labels image
94 x1, y1, x2, y2 -- points delimiting the arc in trigonometric sense
95 cx, cy -- ellipse centre
96 ax, bx -- ellipse axes; ax is the greater axis
97 theta -- orientation of the ellipse
98 ang_start, ang_end -- angles delimiting the arc in trigonometric sense */
99 if( ell_out != NULL )
100 {
101 if( (ell_ascii = fopen("out_ellipse.txt","w")) == NULL )
102 error("main: can't open ellipse output file.");
103 for( i=0; i<ell_count; i++ )
104 {
105 fprintf( ell_ascii,"%d ", ell_labels[i] );
106 fprintf( ell_ascii,"%lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf \n",
107 ell_out[i].x1, ell_out[i].y1, ell_out[i].x2, ell_out[i].y2,
108 ell_out[i].cx, ell_out[i].cy, ell_out[i].ax, ell_out[i].bx,
109 ell_out[i].theta, ell_out[i].ang_start, ell_out[i].ang_end );
110 }
111 fclose(ell_ascii);
112 }
113
114 /* Polygon file: each line contains 2 integers and a variable number of
115 doubles in the form
116 poly_label n x1 y1 x2 y2 ... xn yn
117 where:
118 poly_label -- the label of the pixels supporting this polygon in
119 'out' labels image
120 n -- the number of points in the polygon; it is the double
121 of the number of segments
122 x1 y1 ... xn yn -- (x,y) coordinates of the ending points of each segment.
123 A polygon with n/2 line segments has n points, given in consecutive order.
124 */
125 if( poly_out != NULL )
126 {
127 if( (poly_ascii = fopen("out_polygon.txt","w")) == NULL )
128 error("main: can't open polygon output file.");
129 for( i=0; i<poly_count; i++ )
130 {
131 fprintf( poly_ascii, "%d %d ", poly_labels[i], poly_out[i].dim );
132 for( j=0; j<poly_out[i].dim; j++ )
133 fprintf( poly_ascii,"%lf %lf ", poly_out[i].pts[j].x, poly_out[i].pts[j].y);
134 fprintf( poly_ascii, "\n" );
135 }
136 fclose(poly_ascii);
137 }
138
139 /* write vectorial output in SVG format */
140 if( (ell_out != NULL) || (poly_out != NULL) )
141 {
142 /* init svg file */
143 fsvg = init_svg( "output.svg", xsize, ysize );
144
145 /* write ellipses */
146 for( i=0; i<ell_count; i++)
147 /* distinguish between circle and ellipse, because the procedures
148 to write them are different */
149 if( (double_equal( ell_out[i].ax, ell_out[i].bx )) )/* && (ell_out[i].theta == 0) ) */
150 write_svg_circ_arc( fsvg, &ell_out[i] );
151 else
152 write_svg_ell_arc( fsvg, &ell_out[i] );
153
154 /* write polygons */
155 for( i=0; i<poly_count; i++ )
156 write_svg_poly( fsvg, &poly_out[i] );
157 /* close svg file */
158 fclose_svg( fsvg );
159 }
160
161 /* write labels image in pgm form */
162 write_pgm_image_int( out->data, out->xsize, out->ysize, "labels.pgm" );
163 free_PImageInt(out);
164 if( ell_out != NULL ) {free(ell_out); free(ell_labels);}
165 if( poly_out != NULL )
166 {
167 for( i=0; i<poly_count; i++ )
168 free(poly_out[i].pts);
169 free(poly_out);
170 free(poly_labels);
171 }
172 printf("Number of ellipses detected = %d\n",ell_count);
173 printf("Number of polygons detected = %d\n",poly_count);
174 return 0;
175}
