122/*----------------------------------------------------------------------------*/
126int check_ell( double *eparam )
127{
128 /* check parameters */
129 if( eparam == NULL ) error("check_ell: invalid ellipse.");
130
131 /* reject if degenerate ellipse (negative axes) */
132 if( (eparam[2] <= 1) || (eparam[3] <= 1) ) return FALSE;
133 /* make sure 'ax' is the greater axis */
134 if( eparam[2] < eparam[3] )
135 {
136 swap_double( &(eparam[2]), &(eparam[3]) );
137 eparam[4] += M_1_2_PI;
138 }
139 /* reject directly if very thin ellipse, to win time */
140 if( eparam[2] / eparam[3] > 100 ) return FALSE;
141
142 /* make sure orientation belongs to [-pi/2, pi/2] */
143 if( eparam[4] > M_1_2_PI ) eparam[4] -= M_PI;
144 if( eparam[4] > M_PI ) eparam[4] -= M_PI;
145 if( eparam[4] < -M_1_2_PI ) eparam[4] += M_PI;
146
147 return TRUE;
148}

151/*----------------------------------------------------------------------------*/
156int is_in_circ_ring( Ring * r, int x, int y )
157{
158 /* check parameters */
159 if( r == NULL ) error("is_in_ring: invalid input ring.");
160
161 double d = dist( x, y, r->cx, r->cy) - r->ax;
162
163 double theta = atan2( (double)y - r->cy, (double)x - r->cx);
164
165 /* make sure angle in [0, 2pi] */
166 if( theta<0 ) theta += M_2__PI;
167
168 if( r->ang_end > r->ang_start )
169 {
170 if( (theta < r->ang_start) || (theta > r->ang_end) ) return FALSE;
171 }
172 else
173 if( (theta < r->ang_start) && (theta > r->ang_end) ) return FALSE;
174
175 if( (d < r->wmin) || (d > r->wmax) ) return FALSE;
176
177 return TRUE;
178}

214void rosin_point( Ring *ering, int x, int y, double *xi, double *yi )
215{
216 double xtmp, ytmp;
217 double ae2, be2, fe2;
218 double xp, yp, xp2, yp2, xx, yy;
219 double delta, A, ah, bh2, term;
220 double d[4];
221 int pos;
222 xtmp = x - ering->cx;
223 ytmp = y - ering->cy;
224 ae2 = ering->ax * ering->ax;
225 be2 = ering->bx * ering->bx;
226 fe2 = ae2 - be2;
227 xp = xtmp * cos(-ering->theta) - ytmp * sin(-ering->theta);
228 yp = xtmp * sin(-ering->theta) + ytmp * cos(-ering->theta);
229 xp2 = xp * xp;
230 yp2 = yp * yp;
231 delta = (xp2 + yp2 + fe2) * (xp2 + yp2 + fe2) - 4 * xp2 * fe2;
232 A = (xp2 + yp2 + fe2 - sqrt(delta))/2.0;
233 ah = sqrt(A);
234 bh2 = fe2 - A;
235 term = (A * be2 + ae2 * bh2);
236 xx = ah * sqrt( ae2 * (be2 + bh2) / term );
237 yy = ering->bx * sqrt( bh2 * (ae2 - A) / term );
238
239 d[0] = dist( xp, yp, xx, yy );
240 d[1] = dist( xp, yp, xx,-yy );
241 d[2] = dist( xp, yp,-xx, yy );
242 d[3] = dist( xp, yp,-xx,-yy );
243
244 min_array_pos( d, 4, &pos );
245 switch(pos)
246 {
247 case 0: break;
248 case 1: yy = -yy; break;
249 case 2: xx = -xx; break;
250 case 3: xx = -xx; yy = -yy; break;
251 default: break;
252 }
253 *xi = xx * cos(ering->theta) - yy * sin(ering->theta) + ering->cx;
254 *yi = xx * sin(ering->theta) + yy * cos(ering->theta) + ering->cy;
255}

258/*----------------------------------------------------------------------------*/
262double d_rosin( Ring *ering, double x, double y )
263{
264 double xtmp, ytmp;
265 double ae2, be2, fe2;
266 double xp, yp, xp2, yp2, xx, yy;
267 double delta, A, ah, bh2, term;
268 double d[4];
269
270 xtmp = x - ering->cx;
271 ytmp = y - ering->cy;
272 ae2 = ering->ax * ering->ax;
273 be2 = ering->bx * ering->bx;
274 fe2 = ae2 - be2;
275 xp = xtmp * cos(-ering->theta) - ytmp * sin(-ering->theta);
276 yp = xtmp * sin(-ering->theta) + ytmp * cos(-ering->theta);
277 xp2 = xp * xp;
278 yp2 = yp * yp;
279 delta = (xp2 + yp2 + fe2) * (xp2 + yp2 + fe2) - 4 * xp2 * fe2;
280 A = (xp2 + yp2 + fe2 - sqrt(delta))/2.0;
281 ah = sqrt(A);
282 bh2 = fe2 - A;
283 term = (A * be2 + ae2 * bh2);
284 xx = ah * sqrt( ae2 * (be2 + bh2) / term );
285 yy = ering->bx * sqrt( bh2 * (ae2 - A) / term );
286
287 d[0] = dist( xp, yp, xx, yy);
288 d[1] = dist( xp, yp, xx,-yy);
289 d[2] = dist( xp, yp,-xx, yy);
290 d[3] = dist( xp, yp,-xx,-yy);
291 double dmin = min_array(d,4);
292 if( xp2+yp2 > xx*xx+yy*yy ) return dmin;
293 else return -dmin;
294}

300void ellipse_foci( double *param, double *foci )
301{
302 double f;
303 /* check parameters */
304 if( param == NULL) error("ellipse_foci: invalid input ellipse.");
305 if( foci == NULL) error("ellipse_foci: 'foci' must be non null.");
306
307 f = sqrt( param[2] * param[2] - param[3] * param[3] );
308 foci[0] = param[0] + f * cos(param[4]);
309 foci[1] = param[1] + f * sin(param[4]);
310 foci[2] = param[0] - f * cos(param[4]);
311 foci[3] = param[1] - f * sin(param[4]);
312}

315/*----------------------------------------------------------------------------*/
319double ellipse_normal_angle( double x, double y, double *foci )
320{
321 double tmp1, tmp2, tmp3, theta;
322 /* check parameters */
323 if( foci == NULL ) error("ellipse_normal_angle: 'foci' must be non null");
324
325 tmp1 = atan2( y-foci[1], x-foci[0]);
326 tmp2 = atan2( y-foci[3], x-foci[2]);
327 tmp3 = angle_diff_signed( tmp1, tmp2 );
328
329 theta = tmp1 - tmp3/2.0;
330 while( theta <= -M_PI ) theta += M_2__PI;
331 while( theta > M_PI ) theta -= M_2__PI;
332 return theta;
333}

377static void circ_ring_width( Point *reg, int reg_size, Ring *cring )
378{
379 int i;
380 double w;
381 double wmin = DBL_MAX;
382 double wmax = -DBL_MAX;
383 /* check parameters */
384 if( reg == NULL ) error("circ_ring_width: invalid region.");
385 if( cring == NULL ) error("circ_ring_width: invalid ring.");
386
387 /* compute widths */
388 for( i=0; i<reg_size; i++ )
389 {
390 w = dist( reg[i].x, reg[i].y, cring->cx, cring->cy) - cring->ax;
391 if( w<wmin ) wmin = w;
392 if( w>wmax ) wmax = w;
393 }
394 cring->wmin = wmin; cring->wmax = wmax;
395}

423/*----------------------------------------------------------------------------*/
427int get_ring( Point *reg, int reg_size, double ang, double *param,
428 double *pext1, double *pext2, int conic, double spir, Ring *r,
429 int *grad_dir, double *foci, int msize )
430{
431 double tang;
432 /* check parameters */
433 if( reg == NULL ) error("get_ring: invalid region.");
434 if( param == NULL ) error("get_ring: invalid conic parameters.");
435 if( r == NULL ) error("get_ring: ring must be non null.");
436 if( foci == NULL ) error("get_ring: foci must be non null.");
437
438 /* fill in centre */
439 r->cx = param[0]; r->cy = param[1];
440
441 /* Init 'full' flag; by default, set to 0. */
442 r->full = 0;
443
444 if( conic == CIRCLE )
445 {
446 if( !check_circ(param) ) return FALSE;
447 *grad_dir = dir_gradient_circ( reg[0].x, reg[0].y, ang, param );
448 }
449 else
450 {
451 if( !check_ell(param) ) return FALSE;
452 ellipse_foci( param, foci );
453 *grad_dir = dir_gradient_ell( reg[0].x, reg[0].y, ang, foci );
454 }
455 /* fill in axes and orientation: this must be done after check, as for the
456 ellipse the axes might have been swapped, and orientation normalized */
457 r->ax = param[2]; r->bx = param[3];
458 r->theta = param[4];
459
460 /* fill in extremal points */
461 r->x1 = pext1[0]; r->y1 = pext1[1];
462 r->x2 = pext2[0]; r->y2 = pext2[1];
463
464 /* if gradient converges, interchange the extremal points to keep
465 trigonometric sense */
466 if( *grad_dir == 0 )
467 {
468 swap_double( &(r->x1), &(r->x2) );
469 swap_double( &(r->y1), &(r->y2) );
470 }
471
472 /* set delimiting angles */
473 r->ang_start = atan2( r->y1 - r->cy, r->x1 - r->cx );
474 if( r->ang_start < 0 ) r->ang_start += M_2__PI;
475 r->ang_end = atan2( r->y2 - r->cy, r->x2 - r->cx );
476 if( r->ang_end < 0 ) r->ang_end += M_2__PI;
477
478 /* if the ends of the last rectangles got interwined, swap the ends
479 of the ring */
480 if( r->ang_start > r->ang_end ) tang = r->ang_end + M_2__PI;
481 else tang = r->ang_end;
482 if( (spir > M_PI) && (tang - r->ang_start < M_1_2_PI/2.0) )
483 swap_ring( r );
484
485 /* set ring widths */
486 if( conic == CIRCLE )
487 circ_ring_width( reg, reg_size, r );
488 else
489 ell_ring_width( reg, reg_size, r );
490
491 /* abnormal case: width of the ring is bigger than little axis ->
492 discard ring */
493 if( r->wmax - r->wmin > r->bx ) return FALSE;
494 if( r->ax > msize ) return FALSE;
495
496 return TRUE;
497}
